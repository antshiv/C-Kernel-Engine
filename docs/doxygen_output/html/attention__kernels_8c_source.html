42 // Naive, reference-quality scaled dot-product attention with causal mask.
43 //
44 // Q, K, V are head-major:
45 // q[h, t, d] at q[h * T * aligned_head_dim + t * aligned_head_dim + d]
46 // same for k and v.
47 //
48 // scores buffer must be at least:
49 // num_heads * aligned_context_window * aligned_context_window floats.
50 //
51 // output has same layout as Q/V:
52 // out[h, t, d] at out[h * T * aligned_head_dim + t * aligned_head_dim + d]
53 //
54 // Only the first head_dim elements of each vector participate in the math;
55 // aligned_head_dim allows callers to pad to cache-friendly widths.
56 void attention_forward_causal_head_major(const float *q,
57  const float *k,
58  const float *v,
59  float *scores,
60  float *output,
61  int num_heads,
62  int num_tokens,
63  int head_dim,
64  int aligned_head_dim,
65  int aligned_context_window)
66 {
67  const float scale = 1.0f / sqrtf((float)head_dim);
68 
69  // Phase 1: compute scaled dot-product scores Q·K^T / sqrt(d_k),
70  // lower triangle only (j <= i).
71  for (int h = 0; h < num_heads; ++h) {
72  for (int i = 0; i < num_tokens; ++i) {
73  for (int j = 0; j <= i; ++j) {
74  float dot = 0.0f;
75  size_t base_q = qkv_index(h, i, 0, num_tokens, aligned_head_dim);
76  size_t base_k = qkv_index(h, j, 0, num_tokens, aligned_head_dim);
77 
78  for (int d = 0; d < head_dim; ++d) {
79  dot += q[base_q + d] * k[base_k + d];
80  }
81 
82  scores[score_index(h, i, j, aligned_context_window)] = dot * scale;
83  }
84 
85  // Ensure upper triangle is zeroed so there are no stale values
86  // before the softmax kernel runs.
87  for (int j = i + 1; j < num_tokens; ++j) {
88  scores[score_index(h, i, j, aligned_context_window)] = 0.0f;
89  }
90  }
91  }
92 
93  // Phase 2: apply causal row-wise softmax in-place over j <= i.
94  causal_softmax_head_major(scores,
95  num_heads,
96  num_tokens,
97  aligned_context_window);
98 
99  // Phase 3: attention weights · V.
100  for (int h = 0; h < num_heads; ++h) {
101  for (int i = 0; i < num_tokens; ++i) {
102  size_t out_base = qkv_index(h, i, 0, num_tokens, aligned_head_dim);
103 
104  // Zero the full aligned head slice so padded dims stay clean.
105  for (int d = 0; d < aligned_head_dim; ++d) {
106  output[out_base + d] = 0.0f;
107  }
108 
109  // Weighted sum over causal positions.
110  for (int j = 0; j <= i; ++j) {
111  float w = scores[score_index(h, i, j, aligned_context_window)];
112  size_t v_base = qkv_index(h, j, 0, num_tokens, aligned_head_dim);
113 
114  for (int d = 0; d < head_dim; ++d) {
115  output[out_base + d] += w * v[v_base + d];
116  }
117  }
118  }
119  }
120 }

224 // ============================================================================
225 // ATTENTION FORWARD - Flash-style (no scores materialization)
226 // ============================================================================
227 //
228 // Computes the same causal attention output as `attention_forward_causal_head_major_gqa`,
229 // but does not materialize the [H, T, T] score/weight matrices. This is useful for:
230 // - Prefill: avoids large scratch buffers and improves cache locality
231 // - Decode: supports KV-cache attention for a single token
232 //
233 // Note: This is a correctness-first implementation using an online softmax
234 // accumulator; further blocking/SIMD/AMX optimizations can be layered on later.
235 
236 static void attention_flash_query_causal(const float *q_vec,
237  const float *k_head,
238  const float *v_head,
239  int kv_tokens,
240  int head_dim,
241  int aligned_head_dim,
242  float scale,
243  float *out_vec)
244 {
245  // Online softmax:
246  // m = running max, s = running sum(exp(score - m))
247  // out = sum(exp(score - m) * v)
248  float m = -INFINITY;
249  float s = 0.0f;
250 
251  for (int d = 0; d < head_dim; ++d) {
252  out_vec[d] = 0.0f;
253  }
254 
255  for (int j = 0; j < kv_tokens; ++j) {
256  const float *k_vec = k_head + (size_t)j * (size_t)aligned_head_dim;
257  const float *v_vec = v_head + (size_t)j * (size_t)aligned_head_dim;
258 
259  float dot = 0.0f;
260  for (int d = 0; d < head_dim; ++d) {
261  dot += q_vec[d] * k_vec[d];
262  }
263  float score = dot * scale;
264 
265  if (score > m) {
266  float exp_m = (m == -INFINITY) ? 0.0f : expf(m - score);
267  s *= exp_m;
268  for (int d = 0; d < head_dim; ++d) {
269  out_vec[d] *= exp_m;
270  }
271  s += 1.0f;
272  for (int d = 0; d < head_dim; ++d) {
273  out_vec[d] += v_vec[d];
274  }
275  m = score;
276  } else {
277  float e = expf(score - m);
278  s += e;
279  for (int d = 0; d < head_dim; ++d) {
280  out_vec[d] += e * v_vec[d];
281  }
282  }
283  }
284 
285  float inv_s = 1.0f / s;
286  for (int d = 0; d < head_dim; ++d) {
287  out_vec[d] *= inv_s;
288  }
289  for (int d = head_dim; d < aligned_head_dim; ++d) {
290  out_vec[d] = 0.0f;
291  }
292 }

367 // ============================================================================
368 // ATTENTION BACKWARD - Causal, Head-Major, GQA-aware
369 // ============================================================================
370 //
371 // Backward pass for scaled dot-product attention with causal mask.
372 //
373 // Given:
374 // d_output: gradient from the layer above [num_heads, T, head_dim]
375 // q, k, v: saved activations from forward pass
376 // attn_weights: saved softmax output from forward [num_heads, T, T]
377 //
378 // Computes:
379 // d_q: gradient w.r.t. queries [num_heads, T, head_dim]
380 // d_k: gradient w.r.t. keys [num_kv_heads, T, head_dim]
381 // d_v: gradient w.r.t. values [num_kv_heads, T, head_dim]
382 //
383 // Math derivation:
384 // Forward: scores = Q @ K^T / sqrt(d)
385 // weights = causal_softmax(scores)
386 // output = weights @ V
387 //
388 // Backward through V multiply:
389 // d_weights = d_output @ V^T [H, T, T]
390 // d_v = weights^T @ d_output [H_kv, T, d]
391 //
392 // Backward through softmax:
393 // d_scores = softmax_backward(d_weights, weights)
394 //
395 // Backward through Q @ K^T:
396 // d_q = d_scores @ K / sqrt(d) [H, T, d]
397 // d_k = d_scores^T @ Q / sqrt(d) [H_kv, T, d]
398 //
399 // For GQA: multiple query heads share the same KV head, so we accumulate
400 // gradients from all query heads that map to each KV head.
401 //

460 void attention_backward_causal_head_major_gqa(
461  const float *d_output, // [num_heads, T, aligned_head_dim]
462  const float *q, // [num_heads, T, aligned_head_dim]
463  const float *k, // [num_kv_heads, T, aligned_head_dim]
464  const float *v, // [num_kv_heads, T, aligned_head_dim]
465  const float *attn_weights, // [num_heads, T, aligned_context_window]
466  float *d_q, // [num_heads, T, aligned_head_dim] output
467  float *d_k, // [num_kv_heads, T, aligned_head_dim] output
468  float *d_v, // [num_kv_heads, T, aligned_head_dim] output
469  float *d_scores, // [num_heads, T, aligned_context_window] scratch
470  int num_heads,
471  int num_kv_heads,
472  int num_tokens,
473  int head_dim,
474  int aligned_head_dim,
475  int aligned_context_window)
476 {
477  const float scale = 1.0f / sqrtf((float)head_dim);
478  int T = num_tokens;
479  int H = num_heads;
480  int H_kv = num_kv_heads;
481  int hd = head_dim;
482  int ad = aligned_head_dim;
483  int aw = aligned_context_window;
484 
485  const size_t d_q_elems = (size_t)H * (size_t)T * (size_t)ad;
486  const size_t kv_elems = (size_t)H_kv * (size_t)T * (size_t)ad;
487  /* Zero the aligned outputs so padded lanes never leak garbage to downstream GEMMs. */
488  for (size_t idx = 0; idx < d_q_elems; ++idx) {
489  d_q[idx] = 0.0f;
490  }
491  for (size_t idx = 0; idx < kv_elems; ++idx) {
492  d_k[idx] = 0.0f;
493  d_v[idx] = 0.0f;
494  }
495 
496  // Process each query head
497  for (int h = 0; h < H; ++h) {
498  // Which KV head does this query head use?
499  int kv_h = (int)((long long)h * (long long)H_kv / (long long)H);
500 
501  // ----------------------------------------------------------------
502  // Step 1: d_weights = d_output @ V^T and d_v += weights^T @ d_output
503  // ----------------------------------------------------------------
504  // For each query position i, compute d_weights[i, j] for j <= i
505  // and accumulate d_v[j] contributions
506 
507  for (int i = 0; i < T; ++i) {
508  size_t d_out_base = qkv_index(h, i, 0, T, ad);
509 
510  for (int j = 0; j <= i; ++j) {
511  size_t v_base = qkv_index(kv_h, j, 0, T, ad);
512  size_t w_idx = score_index(h, i, j, aw);
513  float w = attn_weights[w_idx];
514 
515  // d_weights[h, i, j] = d_output[h, i, :] @ v[kv_h, j, :]^T
516  float dot = 0.0f;
517  for (int dd = 0; dd < hd; ++dd) {
518  dot += d_output[d_out_base + dd] * v[v_base + dd];
519  }
520  d_scores[w_idx] = dot;
521 
522  // d_v[kv_h, j, :] += weights[h, i, j] * d_output[h, i, :]
523  for (int dd = 0; dd < hd; ++dd) {
524  d_v[v_base + dd] += w * d_output[d_out_base + dd];
525  }
526  }
527 
528  // Zero out upper triangle of d_scores
529  for (int j = i + 1; j < T; ++j) {
530  d_scores[score_index(h, i, j, aw)] = 0.0f;
531  }
532  /* Scores scratch uses aligned_context_window, zero the padded columns. */
533  for (int j = T; j < aw; ++j) {
534  d_scores[score_index(h, i, j, aw)] = 0.0f;
535  }
536  }
537 
538  // ----------------------------------------------------------------
539  // Step 2: Backward through softmax (in-place on d_scores for this head)
540  // ----------------------------------------------------------------
541  // d_scores = softmax_backward(d_scores, attn_weights)
542  // Formula: d_score[i,j] = w[i,j] * (d_w[i,j] - sum_k(w[i,k] * d_w[i,k]))
543 
544  for (int i = 0; i < T; ++i) {
545  int base = h * aw * aw + i * aw;
546 
547  // Compute dot product: sum_j w[i,j] * d_w[i,j]
548  float dot_product = 0.0f;
549  for (int j = 0; j <= i; ++j) {
550  float wt = attn_weights[base + j];
551  float dw = d_scores[base + j];
552  dot_product += wt * dw;
553  }
554 
555  // Apply softmax backward formula
556  for (int j = 0; j <= i; ++j) {
557  float wt = attn_weights[base + j];
558  float dw = d_scores[base + j];
559  d_scores[base + j] = wt * (dw - dot_product);
560  }
561  }
562 
563  // ----------------------------------------------------------------
564  // Step 3: d_q = d_scores @ K * scale
565  // d_k += d_scores^T @ Q * scale
566  // ----------------------------------------------------------------
567 
568  for (int i = 0; i < T; ++i) {
569  size_t d_q_base = qkv_index(h, i, 0, T, ad);
570  size_t q_base = qkv_index(h, i, 0, T, ad);
571
