28 // Naive, reference-quality scaled dot-product attention with causal mask.
29 //
30 // Q, K, V are head-major:
31 // q[h, t, d] at q[h * T * aligned_head_dim + t * aligned_head_dim + d]
32 // same for k and v.
33 //
34 // scores buffer must be at least:
35 // num_heads * aligned_context_window * aligned_context_window floats.
36 //
37 // output has same layout as Q/V:
38 // out[h, t, d] at out[h * T * aligned_head_dim + t * aligned_head_dim + d]
39 //
40 // Only the first head_dim elements of each vector participate in the math;
41 // aligned_head_dim allows callers to pad to cache-friendly widths.
42 void attention_forward_causal_head_major(const float *q,
43  const float *k,
44  const float *v,
45  float *scores,
46  float *output,
47  int num_heads,
48  int num_tokens,
49  int head_dim,
50  int aligned_head_dim,
51  int aligned_context_window)
52 {
53  const float scale = 1.0f / sqrtf((float)head_dim);
54 
55  // Phase 1: compute scaled dot-product scores Q·K^T / sqrt(d_k),
56  // lower triangle only (j <= i).
57  for (int h = 0; h < num_heads; ++h) {
58  for (int i = 0; i < num_tokens; ++i) {
59  for (int j = 0; j <= i; ++j) {
60  float dot = 0.0f;
61  size_t base_q = qkv_index(h, i, 0, num_tokens, aligned_head_dim);
62  size_t base_k = qkv_index(h, j, 0, num_tokens, aligned_head_dim);
63 
64  for (int d = 0; d < head_dim; ++d) {
65  dot += q[base_q + d] * k[base_k + d];
66  }
67 
68  scores[score_index(h, i, j, aligned_context_window)] = dot * scale;
69  }
70 
71  // Ensure upper triangle is zeroed so there are no stale values
72  // before the softmax kernel runs.
73  for (int j = i + 1; j < num_tokens; ++j) {
74  scores[score_index(h, i, j, aligned_context_window)] = 0.0f;
75  }
76  }
77  }
78 
79  // Phase 2: apply causal row-wise softmax in-place over j <= i.
80  causal_softmax_head_major(scores,
81  num_heads,
82  num_tokens,
83  aligned_context_window);
84 
85  // Phase 3: attention weights · V.
86  for (int h = 0; h < num_heads; ++h) {
87  for (int i = 0; i < num_tokens; ++i) {
88  size_t out_base = qkv_index(h, i, 0, num_tokens, aligned_head_dim);
89 
90  // Zero the full aligned head slice so padded dims stay clean.
91  for (int d = 0; d < aligned_head_dim; ++d) {
92  output[out_base + d] = 0.0f;
93  }
94 
95  // Weighted sum over causal positions.
96  for (int j = 0; j <= i; ++j) {
97  float w = scores[score_index(h, i, j, aligned_context_window)];
98  size_t v_base = qkv_index(h, j, 0, num_tokens, aligned_head_dim);
99 
100  for (int d = 0; d < head_dim; ++d) {
101  output[out_base + d] += w * v[v_base + d];
102  }
103  }
104  }
105  }
106 }

170 // ============================================================================
171 // ATTENTION BACKWARD - Causal, Head-Major, GQA-aware
172 // ============================================================================
173 //
174 // Backward pass for scaled dot-product attention with causal mask.
175 //
176 // Given:
177 // d_output: gradient from the layer above [num_heads, T, head_dim]
178 // q, k, v: saved activations from forward pass
179 // attn_weights: saved softmax output from forward [num_heads, T, T]
180 //
181 // Computes:
182 // d_q: gradient w.r.t. queries [num_heads, T, head_dim]
183 // d_k: gradient w.r.t. keys [num_kv_heads, T, head_dim]
184 // d_v: gradient w.r.t. values [num_kv_heads, T, head_dim]
185 //
186 // Math derivation:
187 // Forward: scores = Q @ K^T / sqrt(d)
188 // weights = causal_softmax(scores)
189 // output = weights @ V
190 //
191 // Backward through V multiply:
192 // d_weights = d_output @ V^T [H, T, T]
193 // d_v = weights^T @ d_output [H_kv, T, d]
194 //
195 // Backward through softmax:
196 // d_scores = softmax_backward(d_weights, weights)
197 //
198 // Backward through Q @ K^T:
199 // d_q = d_scores @ K / sqrt(d) [H, T, d]
200 // d_k = d_scores^T @ Q / sqrt(d) [H_kv, T, d]
201 //
202 // For GQA: multiple query heads share the same KV head, so we accumulate
203 // gradients from all query heads that map to each KV head.
204 //
205 void attention_backward_causal_head_major_gqa(
206  const float *d_output, // [num_heads, T, aligned_head_dim]
207  const float *q, // [num_heads, T, aligned_head_dim]
208  const float *k, // [num_kv_heads, T, aligned_head_dim]
209  const float *v, // [num_kv_heads, T, aligned_head_dim]
210  const float *attn_weights, // [num_heads, T, aligned_context_window]
211  float *d_q, // [num_heads, T, aligned_head_dim] output
212  float *d_k, // [num_kv_heads, T, aligned_head_dim] output
213  float *d_v, // [num_kv_heads, T, aligned_head_dim] output
214  float *d_scores, // [num_heads, T, aligned_context_window] scratch
215  int num_heads,
216  int num_kv_heads,
217  int num_tokens,
218  int head_dim,
219  int aligned_head_dim,
220  int aligned_context_window)
221 {
222  const float scale = 1.0f / sqrtf((float)head_dim);
223  int T = num_tokens;
224  int H = num_heads;
225  int H_kv = num_kv_heads;
226  int hd = head_dim;
227  int ad = aligned_head_dim;
228  int aw = aligned_context_window;
229 
230  // Zero d_k and d_v (we accumulate from multiple query heads for GQA)
231  for (int kv_h = 0; kv_h < H_kv; ++kv_h) {
232  for (int t = 0; t < T; ++t) {
233  size_t base = qkv_index(kv_h, t, 0, T, ad);
234  for (int i = 0; i < hd; ++i) {
235  d_k[base + i] = 0.0f;
236  d_v[base + i] = 0.0f;
237  }
238  }
239  }
240 
241  // Process each query head
242  for (int h = 0; h < H; ++h) {
243  // Which KV head does this query head use?
244  int kv_h = (int)((long long)h * (long long)H_kv / (long long)H);
245 
246  // ----------------------------------------------------------------
247  // Step 1: d_weights = d_output @ V^T and d_v += weights^T @ d_output
248  // ----------------------------------------------------------------
249  // For each query position i, compute d_weights[i, j] for j <= i
250  // and accumulate d_v[j] contributions
251 
252  for (int i = 0; i < T; ++i) {
253  size_t d_out_base = qkv_index(h, i, 0, T, ad);
254 
255  for (int j = 0; j <= i; ++j) {
256  size_t v_base = qkv_index(kv_h, j, 0, T, ad);
257  size_t w_idx = score_index(h, i, j, aw);
258  float w = attn_weights[w_idx];
259 
260  // d_weights[h, i, j] = d_output[h, i, :] @ v[kv_h, j, :]^T
261  float dot = 0.0f;
262  for (int dd = 0; dd < hd; ++dd) {
263  dot += d_output[d_out_base + dd] * v[v_base + dd];
264  }
265  d_scores[w_idx] = dot;
266 
267  // d_v[kv_h, j, :] += weights[h, i, j] * d_output[h, i, :]
268  for (int dd = 0; dd < hd; ++dd) {
269  d_v[v_base + dd] += w * d_output[d_out_base + dd];
270  }
271  }
272 
273  // Zero out upper triangle of d_scores
274  for (int j = i + 1; j < T; ++j) {
275  d_scores[score_index(h, i, j, aw)] = 0.0f;
276  }
277  }
278 
279  // ----------------------------------------------------------------
280  // Step 2: Backward through softmax (in-place on d_scores for this head)
281  // ----------------------------------------------------------------
282  // d_scores = softmax_backward(d_scores, attn_weights)
283  // Formula: d_score[i,j] = w[i,j] * (d_w[i,j] - sum_k(w[i,k] * d_w[i,k]))
284 
285  for (int i = 0; i < T; ++i) {
286  int base = h * aw * aw + i * aw;
287 
288  // Compute dot product: sum_j w[i,j] * d_w[i,j]
289  float dot_product = 0.0f;
290  for (int j = 0; j <= i; ++j) {
291  float wt = attn_weights[base + j];
292  float dw = d_scores[base + j];
293  dot_product += wt * dw;
294  }
295 
296  // Apply softmax backward formula
297  for (int j = 0; j <= i; ++j) {
298  float wt = attn_weights[base + j];
299  float dw = d_scores[base + j];
300  d_scores[base + j] = wt * (dw - dot_product);
301  }
302  }
303 
304  // ----------------------------------------------------------------
305  // Step 3: d_q = d_scores @ K * scale
306  // d_k += d_scores^T @ Q * scale
307  // ----------------------------------------------------------------
308 
309  for (int i = 0; i < T; ++i) {
310  size_t d_q_base = qkv_index(h, i, 0, T, ad);
311  size_t q_base = qkv_index(h, i, 0, T, ad);
312 
313  // Zero d_q for this position
314  for (int dd = 0; dd < hd; ++dd) {
315  d_q[d_q_base + dd] = 0.0f;
316  }
317 
318  // d_q[h, i, :] = sum_j d_scores[h, i, j] * k[kv_h, j, :] * scale
319  // d_k[kv_h, j, :] += d_scores[h, i, j] * q[h, i, :] * scale
320  for (int j = 0; j <= i; ++j) {
321  size_t k_base = qkv_index(kv_h, j, 0, T, ad);
322  size_t d_k_base = qkv_index(kv_h, j, 0, T, ad);
323  float ds = d_scores[score_index(h, i, j, aw)] * scale;
324 
325  for (int dd = 0; dd < hd; ++dd) {
326  d_q[d_q_base + dd] += ds * k[k_base + dd];
327  d_k[d_k_base + dd] += ds * q[q_base + dd];
328  }
329  }
330  }
331  }
332 }
