1 /**
2  * @file gemm_kernels_q4_0.c
3  * @brief GEMM/GEMV kernels with Q4_0 quantized weights
4  *
5  * Q4_0 Format:
6  * - 32 weights per block
7  * - 1 FP16 scale per block
8  * - 18 bytes per 32 weights = 4.5 bits/weight
9  *
10  * Operations:
11  * Forward: Y = W @ X (W is Q4_0, X and Y are FP32)
12  * Backward: dX = W^T @ dY (gradient w.r.t. input)
13  *
14  * Note: Weight gradients are not computed for quantized weights.
15  * For fine-tuning, use LoRA adapters which maintain FP32 gradients separately.
16  */

31 /**
32  * @brief Matrix-vector multiply with Q4_0 weights (scalar reference)
33  *
34  * @param y Output vector [M]
35  * @param W Weight matrix in Q4_0 format [M x K]
36  * @param x Input vector [K]
37  * @param M Number of output rows
38  * @param K Number of columns (must be multiple of 32)
39  */
40 void gemv_q4_0_ref(float *y,
41  const void *W,
42  const float *x,
43  int M, int K)
44 {
45  const block_q4_0 *blocks = (const block_q4_0 *)W;
46  const int blocks_per_row = K / QK4_0;
47 
48  for (int row = 0; row < M; row++) {
49  float sum = 0.0f;
50 
51  for (int b = 0; b < blocks_per_row; b++) {
52  const block_q4_0 *block = &blocks[row * blocks_per_row + b];
53  const float d = CK_FP16_TO_FP32(block->d);
54  const float *xp = &x[b * QK4_0];
55 
56  for (int i = 0; i < QK4_0 / 2; i++) {
57  const uint8_t packed = block->qs[i];
58  const int8_t q0 = (packed & 0x0F) - 8;
59  const int8_t q1 = (packed >> 4) - 8;
60 
61  sum += d * (float)q0 * xp[2*i + 0];
62  sum += d * (float)q1 * xp[2*i + 1];
63  }
64  }
65 
66  y[row] = sum;
67  }
68 }

72  * @brief Matrix-vector multiply with Q4_0 weights (AVX-512)
73  */
74 void gemv_q4_0_avx512(float *y,
75  const void *W,
76  const float *x,
77  int M, int K)
78 {
79  const block_q4_0 *blocks = (const block_q4_0 *)W;
80  const int blocks_per_row = K / QK4_0;
81  const __m512i offset = _mm512_set1_epi32(8);
82  const __m512i mask_lo = _mm512_set1_epi32(0x0F);
83 
84  for (int row = 0; row < M; row++) {
85  __m512 acc = _mm512_setzero_ps();
86 
87  for (int b = 0; b < blocks_per_row; b++) {
88  const block_q4_0 *block = &blocks[row * blocks_per_row + b];
89  const __m512 vscale = _mm512_set1_ps(CK_FP16_TO_FP32(block->d));
90  const float *xp = &x[b * QK4_0];
91 
92  /* Load 16 bytes = 32 x 4-bit weights */
93  __m128i packed = _mm_loadu_si128((const __m128i *)block->qs);
94  __m512i bytes = _mm512_cvtepu8_epi32(packed);
95 
96  /* Extract and dequantize */
97  __m512i lo = _mm512_sub_epi32(_mm512_and_epi32(bytes, mask_lo), offset);
98  __m512i hi = _mm512_sub_epi32(_mm512_srli_epi32(bytes, 4), offset);
99 
100  __m512 w_lo = _mm512_mul_ps(_mm512_cvtepi32_ps(lo), vscale);
101  __m512 w_hi = _mm512_mul_ps(_mm512_cvtepi32_ps(hi), vscale);
102 
103  /* Load interleaved input */
104  __m512 x_even = _mm512_set_ps(
105  xp[30], xp[28], xp[26], xp[24], xp[22], xp[20], xp[18], xp[16],
106  xp[14], xp[12], xp[10], xp[8], xp[6], xp[4], xp[2], xp[0]);
107  __m512 x_odd = _mm512_set_ps(
108  xp[31], xp[29], xp[27], xp[25], xp[23], xp[21], xp[19], xp[17],
109  xp[15], xp[13], xp[11], xp[9], xp[7], xp[5], xp[3], xp[1]);
110 
111  acc = _mm512_fmadd_ps(w_lo, x_even, acc);
112  acc = _mm512_fmadd_ps(w_hi, x_odd, acc);
113  }
114 
115  y[row] = _mm512_reduce_add_ps(acc);
116  }
117 }

152 /* ============================================================================
153  * Backward Pass: Gradient w.r.t. Input
154  *
155  * Given: dL/dY (gradient of loss w.r.t. output)
156  * Compute: dL/dX = W^T @ dL/dY
157  *
158  * For quantized weights, we dequantize on-the-fly during backprop.
159  * Weight gradients are NOT computed (weights are frozen).
160  * ============================================================================ */
161 
162 /**
163  * @brief Backward pass: compute input gradient
164  *
165  * @param dX Output gradient w.r.t. input [K]
166  * @param W Weight matrix in Q4_0 format [M x K]
167  * @param dY Gradient w.r.t. output [M]
168  * @param M Number of output rows
169  * @param K Number of columns (input dimension)
170  */
171 void gemv_q4_0_backward_ref(float *dX,
172  const void *W,
173  const float *dY,
174  int M, int K)
175 {
176  const block_q4_0 *blocks = (const block_q4_0 *)W;
177  const int blocks_per_row = K / QK4_0;
178 
179  /* Zero output gradient */
180  memset(dX, 0, K * sizeof(float));
181 
182  /* Accumulate: dX += W^T @ dY */
183  for (int row = 0; row < M; row++) {
184  const float dy = dY[row];
185 
186  for (int b = 0; b < blocks_per_row; b++) {
187  const block_q4_0 *block = &blocks[row * blocks_per_row + b];
188  const float d = CK_FP16_TO_FP32(block->d);
189  float *dxp = &dX[b * QK4_0];
190 
191  for (int i = 0; i < QK4_0 / 2; i++) {
192  const uint8_t packed = block->qs[i];
193  const int8_t q0 = (packed & 0x0F) - 8;
194  const int8_t q1 = (packed >> 4) - 8;
195 
196  dxp[2*i + 0] += d * (float)q0 * dy;
197  dxp[2*i + 1] += d * (float)q1 * dy;
198  }
199  }
200  }
201 }
