3  * @brief GEMM/GEMV kernels with Q4_K quantized weights
4  *
5  * Implements matrix multiplication where:
6  * - Activations (input): FP32
7  * - Weights: Q4_K (4.5 bits/weight, nested scales)

9  *

11  * registers and immediately used in FMA, never written to memory.

14  * - gemv_q4_k: Matrix-vector multiply (batch=1, token generation)
15  * - gemm_q4_k: Matrix-matrix multiply (batch>1, prefill)
16  */

28  * GEMV: y = W @ x (W is Q4_K, x and y are FP32)
29  *
30  * For token generation (batch=1), this is the critical path.
31  * Memory-bound: we're loading ~4GB of weights for a 7B model per token.

34 /**
35  * @brief Matrix-vector multiply with Q4_K weights (scalar reference)
36  *
37  * @param y Output vector [M]
38  * @param W Weight matrix in Q4_K format [M x K], stored row-major
39  * @param x Input vector [K]
40  * @param M Number of output rows
41  * @param K Number of columns (must be multiple of 256)
42  */
43 void gemv_q4_k_ref(float *y,
44  const void *W,
45  const float *x,
46  int M, int K)
47 {
48  const block_q4_K *blocks = (const block_q4_K *)W;
49  const int blocks_per_row = K / QK_K; /* QK_K = 256 */
50 
51  for (int row = 0; row < M; row++) {
52  float sum = 0.0f;
53 
54  for (int b = 0; b < blocks_per_row; b++) {
55  const block_q4_K *block = &blocks[row * blocks_per_row + b];
56  const float d = GGML_FP16_TO_FP32(block->d);
57  const float dmin = GGML_FP16_TO_FP32(block->dmin);
58 
59  /* Unpack sub-block scales */
60  uint8_t sc[8], m[8];
61  unpack_q4_k_scales(block->scales, sc, m);
62 
63  /* Process 8 sub-blocks of 32 weights each */
64  for (int sub = 0; sub < 8; sub++) {
65  const float scale = d * (float)sc[sub];
66  const float min_val = dmin * (float)m[sub];
67  const uint8_t *qs = &block->qs[sub * 16];
68  const float *xp = &x[b * QK_K + sub * 32];
69 
70  for (int i = 0; i < 16; i++) {
71  const uint8_t packed = qs[i];
72  const int8_t q0 = (packed & 0x0F) - 8;
73  const int8_t q1 = (packed >> 4) - 8;
74 
75  const float w0 = scale * (float)q0 + min_val;
76  const float w1 = scale * (float)q1 + min_val;
77 
78  sum += w0 * xp[2*i + 0];
79  sum += w1 * xp[2*i + 1];
80  }
81  }
82  }
83 
84  y[row] = sum;
85  }
86 }

89 /**
90  * @brief Matrix-vector multiply with Q4_K weights (AVX-512 optimized)
91  *
92  * Fused dequant + FMA: weights dequantized in ZMM registers, never touch RAM.
93  */
94 void gemv_q4_k_avx512(float *y,
95  const void *W,
96  const float *x,
97  int M, int K)
98 {
99  const block_q4_K *blocks = (const block_q4_K *)W;
100  const int blocks_per_row = K / QK_K;
101 
102  for (int row = 0; row < M; row++) {
103  __m512 acc = _mm512_setzero_ps();
104 
105  for (int b = 0; b < blocks_per_row; b++) {
106  const block_q4_K *block = &blocks[row * blocks_per_row + b];
107  const float d = GGML_FP16_TO_FP32(block->d);
108  const float dmin = GGML_FP16_TO_FP32(block->dmin);
109 
110  uint8_t sc[8], m_arr[8];
111  unpack_q4_k_scales(block->scales, sc, m_arr);
112 
113  /* Process 8 sub-blocks */
114  for (int sub = 0; sub < 8; sub++) {
115  const float scale = d * (float)sc[sub];
116  const float min_val = dmin * (float)m_arr[sub];
117  const __m512 vscale = _mm512_set1_ps(scale);
118  const __m512 vmin = _mm512_set1_ps(min_val);
119  const __m512i offset = _mm512_set1_epi32(8);
120  const __m512i mask_lo = _mm512_set1_epi32(0x0F);
121 
122  const uint8_t *qs = &block->qs[sub * 16];
123  const float *xp = &x[b * QK_K + sub * 32];
124 
125  /* Load 16 bytes = 32 x 4-bit weights */
126  __m128i packed = _mm_loadu_si128((const __m128i *)qs);
127  __m512i bytes = _mm512_cvtepu8_epi32(packed);
128 
129  /* Extract lower nibbles (weights at even indices) */
130  __m512i lo = _mm512_and_epi32(bytes, mask_lo);
131  lo = _mm512_sub_epi32(lo, offset);
132  __m512 w_lo = _mm512_fmadd_ps(_mm512_cvtepi32_ps(lo), vscale, vmin);
133 
134  /* Extract upper nibbles (weights at odd indices) */
135  __m512i hi = _mm512_srli_epi32(bytes, 4);
136  hi = _mm512_sub_epi32(hi, offset);
137  __m512 w_hi = _mm512_fmadd_ps(_mm512_cvtepi32_ps(hi), vscale, vmin);
138 
139  /* Load input vectors */
140  /* Note: weights are interleaved (lo=even indices, hi=odd indices)
141  * We need to load x accordingly */
142  __m512 x_even = _mm512_set_ps(
143  xp[30], xp[28], xp[26], xp[24], xp[22], xp[20], xp[18], xp[16],
144  xp[14], xp[12], xp[10], xp[8], xp[6], xp[4], xp[2], xp[0]);
145  __m512 x_odd = _mm512_set_ps(
146  xp[31], xp[29], xp[27], xp[25], xp[23], xp[21], xp[19], xp[17],
147  xp[15], xp[13], xp[11], xp[9], xp[7], xp[5], xp[3], xp[1]);
148 
149  /* FMA: acc += w * x */
150  acc = _mm512_fmadd_ps(w_lo, x_even, acc);
151  acc = _mm512_fmadd_ps(w_hi, x_odd, acc);
152  }
153  }
154 
155  /* Horizontal sum */
156  y[row] = _mm512_reduce_add_ps(acc);
157  }
158 }
159 #endif /* __AVX512F__ */

369 /* ============================================================================
370  * Backward Pass: Gradient w.r.t. Input
371  *
372  * Given: dL/dY (gradient of loss w.r.t. output)
373  * Compute: dL/dX = W^T @ dL/dY
374  *
375  * For quantized weights, we dequantize on-the-fly during backprop.
376  * Weight gradients are NOT computed (weights are frozen).
377  * For fine-tuning, use LoRA adapters which maintain FP32 gradients separately.
378  * ============================================================================ */
379 
380 /**
381  * @brief Backward pass: compute input gradient (scalar reference)
382  *
383  * @param dX Output gradient w.r.t. input [K]
384  * @param W Weight matrix in Q4_K format [M x K]
385  * @param dY Gradient w.r.t. output [M]
386  * @param M Number of output rows
387  * @param K Number of columns (input dimension)
388  */
389 void gemv_q4_k_backward_ref(float *dX,
390  const void *W,
391  const float *dY,
392  int M, int K)
393 {
394  const block_q4_K *blocks = (const block_q4_K *)W;
395  const int blocks_per_row = K / QK_K;
396 
397  /* Zero output gradient */
398  memset(dX, 0, K * sizeof(float));
399 
400  /* Accumulate: dX += W^T @ dY */
401  for (int row = 0; row < M; row++) {
402  const float dy = dY[row];
403 
404  for (int b = 0; b < blocks_per_row; b++) {
405  const block_q4_K *block = &blocks[row * blocks_per_row + b];
406  const float d = CK_FP16_TO_FP32(block->d);
407  const float dmin = CK_FP16_TO_FP32(block->dmin);
408 
409  uint8_t sc[8], m[8];
410  unpack_q4_k_scales(block->scales, sc, m);
411 
412  for (int sub = 0; sub < 8; sub++) {
413  const float scale = d * (float)sc[sub];
414  const float min_val = dmin * (float)m[sub];
415  const uint8_t *qs = &block->qs[sub * 16];
416  float *dxp = &dX[b * QK_K + sub * 32];
417 
418  for (int i = 0; i < 16; i++) {
419  const uint8_t packed = qs[i];
420  const int8_t q0 = (packed & 0x0F) - 8;
421  const int8_t q1 = (packed >> 4) - 8;
422 
423  const float w0 = scale * (float)q0 + min_val;
424  const float w1 = scale * (float)q1 + min_val;
425 
426  dxp[2*i + 0] += w0 * dy;
427  dxp[2*i + 1] += w1 * dy;
428  }
429  }
430  }
431  }
432 }
