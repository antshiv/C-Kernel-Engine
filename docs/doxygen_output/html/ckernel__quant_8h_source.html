58 /* ============================================================================
59  * Q4_K: K-Quant 4-bit with Nested Scales (Primary Target)
60  * - 256 weights per super-block
61  * - 8 sub-blocks of 32 weights each
62  * - Two-level scaling: super-block FP16 + sub-block 6-bit
63  * - 144 bytes per 256 weights = 4.5 bits/weight
64  *
65  * This is the format used by Q4_K_M, Q4_K_S, Q4_K_L variants.
66  * The M/S/L suffix indicates quantization aggressiveness, not structure.
67  * ============================================================================ */

149 /* ============================================================================
150  * Q4_K Scale Unpacking Utilities
151  *
152  * The scales[12] array packs 8 scales and 8 mins in 6-bit format.
153  * Unpacking is non-trivial due to the bit packing.
154  * ============================================================================ */
155 
156 /**
157  * @brief Unpack Q4_K sub-block scales and mins
158  *
159  * @param scales The packed scales[12] array from block_q4_K
160  * @param sc Output: 8 unpacked scale values (multiply by super-block d)
161  * @param m Output: 8 unpacked min values (multiply by super-block dmin)
162  */
163 static inline void unpack_q4_k_scales(const uint8_t *scales,
164  uint8_t *sc, uint8_t *m) {
165  /*
166  * The 12-byte scales array encodes 8 scales and 8 mins in 6-bit each.
167  * Each 6-bit value is split across byte boundaries.
168  */
169 
170  /* Unpack scales (sc[0..7]) from bytes 0-5 */
171  sc[0] = (scales[0] & 0x3F);
172  sc[1] = (scales[0] >> 6) | ((scales[1] & 0x0F) << 2);
173  sc[2] = (scales[1] >> 4) | ((scales[2] & 0x03) << 4);
174  sc[3] = (scales[2] >> 2);
175  sc[4] = (scales[3] & 0x3F);
176  sc[5] = (scales[3] >> 6) | ((scales[4] & 0x0F) << 2);
177  sc[6] = (scales[4] >> 4) | ((scales[5] & 0x03) << 4);
178  sc[7] = (scales[5] >> 2);
179 
180  /* Unpack mins (m[0..7]) from bytes 6-11 */
181  m[0] = (scales[6] & 0x3F);
182  m[1] = (scales[6] >> 6) | ((scales[7] & 0x0F) << 2);
183  m[2] = (scales[7] >> 4) | ((scales[8] & 0x03) << 4);
184  m[3] = (scales[8] >> 2);
185  m[4] = (scales[9] & 0x3F);
186  m[5] = (scales[9] >> 6) | ((scales[10] & 0x0F) << 2);
187  m[6] = (scales[10] >> 4) | ((scales[11] & 0x03) << 4);
188  m[7] = (scales[11] >> 2);
189 }

195 /**
196  * @brief Convert FP16 (ck_half) to FP32
197  */
198 static inline float ck_fp16_to_fp32(ck_half h) {
199  /*
200  * FP16: 1 sign + 5 exponent + 10 mantissa
201  * FP32: 1 sign + 8 exponent + 23 mantissa
202  */
203  uint32_t sign = (h & 0x8000) << 16;
204  uint32_t exp = (h >> 10) & 0x1F;
205  uint32_t mant = h & 0x3FF;
206 
207  uint32_t result;
208 
209  if (exp == 0) {
210  if (mant == 0) {
211  /* Zero */
212  result = sign;
213  } else {
214  /* Denormalized - convert to normalized FP32 */
215  exp = 1;
216  while ((mant & 0x400) == 0) {
217  mant <<= 1;
218  exp--;
219  }
220  mant &= 0x3FF;
221  result = sign | ((exp + 127 - 15) << 23) | (mant << 13);
222  }
223  } else if (exp == 31) {
224  /* Inf or NaN */
225  result = sign | 0x7F800000 | (mant << 13);
226  } else {
227  /* Normalized */
228  result = sign | ((exp + 127 - 15) << 23) | (mant << 13);
229  }
230 
231  union { uint32_t u; float f; } u;
232  u.u = result;
233  return u.f;
234 }

236 /**
237  * @brief Convert FP32 to FP16 (ck_half)
238  */
239 static inline ck_half ck_fp32_to_fp16(float f) {
240  union { uint32_t u; float f; } u;
241  u.f = f;
242 
243  uint32_t sign = (u.u >> 16) & 0x8000;
244  int32_t exp = ((u.u >> 23) & 0xFF) - 127 + 15;
245  uint32_t mant = (u.u >> 13) & 0x3FF;
246 
247  if (exp <= 0) {
248  if (exp < -10) {
249  /* Underflow to zero */
250  return sign;
251  }
252  /* Denormalized */
253  mant = (mant | 0x400) >> (1 - exp);
254  return sign | mant;
255  } else if (exp >= 31) {
256  /* Overflow to infinity */
257  return sign | 0x7C00;
258  }
259 
260  return sign | (exp << 10) | mant;
261 }
