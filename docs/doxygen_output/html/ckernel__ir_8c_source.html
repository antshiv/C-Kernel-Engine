259 {

261  return -1;
262  }

265  }

268  CKIRNode *nodes = (CKIRNode *)calloc((size_t)N, sizeof(CKIRNode));
269  if (!nodes) {
270  return -1;
271  }
272

275  CKIRNode *b = &nodes[i];
276

278  b->op = map_forward_to_backward(f->op);
279  b->n_inputs = f->n_outputs; // placeholder
280  b->n_outputs= f->n_inputs; // placeholder
281 
282  // For now we simply copy the forward inputs to keep a reference
283  // to which activations this backward op relates to.
284  for (int j = 0; j < f->n_inputs; ++j) {
285  b->inputs[j] = f->inputs[j];
286  }
287  }
288 
289  backward->config = forward->config;
290  backward->num_nodes = N;
291  backward->nodes = nodes;

472  return -1;
473  }
474 
475  if (fseek(f, 0, SEEK_END) != 0) {
476  fclose(f);
477  return -1;
478  }

481  fclose(f);
482  return -1;
483  }
484  if (fseek(f, 0, SEEK_SET) != 0) {
485  fclose(f);
486  return -1;
487  }

490  if (!buf) {
491  fclose(f);
492  return -1;
493  }
494  size_t nread = fread(buf, 1, (size_t)len, f);
495  fclose(f);
496  buf[nread] = '\0';
497 
498  CKIRGraph tmp;
499  memset(&tmp, 0, sizeof(tmp));
500 
501  // Parse config using the same helper as HF-style JSON.
502  if (parse_int_field(buf, "\"num_layers\"", &tmp.config.num_layers) != 0) {
503  fprintf(stderr, "ck_ir_parse_json: missing num_layers\n");
504  }
505  if (parse_int_field(buf, "\"hidden_size\"", &tmp.config.hidden_size) != 0) {
506  fprintf(stderr, "ck_ir_parse_json: missing hidden_size\n");
507  }

509  fprintf(stderr, "ck_ir_parse_json: missing intermediate_size\n");
510  }
511  if (parse_int_field(buf, "\"num_attention_heads\"", &tmp.config.num_heads) != 0) {
512  fprintf(stderr, "ck_ir_parse_json: missing num_attention_heads\n");
513  }
514  if (parse_int_field(buf, "\"num_key_value_heads\"", &tmp.config.num_kv_heads) != 0) {
515  tmp.config.num_kv_heads = tmp.config.num_heads;
516  }
517 
518  // Optional: vocab_size / context_window may be present
519  if (parse_int_field(buf, "\"vocab_size\"", &tmp.config.vocab_size) != 0) {
520  tmp.config.vocab_size = 0;
521  }
522  if (parse_int_field(buf, "\"context_window\"", &tmp.config.context_window) != 0) {
523  tmp.config.context_window = 0;
524  }
525 
526  // Count nodes by scanning for "layer" keys in the nodes array.
527  char *nodes_begin = strstr(buf, "\"nodes\"");
528  if (!nodes_begin) {
529  free(buf);
530  return -1;
531  }
532  char *p = nodes_begin;
533  int count = 0;
534  while ((p = strstr(p, "\"layer\"")) != NULL) {
535  count++;
536  p += 7;
537  }
538  if (count <= 0) {
539  free(buf);
540  return -1;
541  }
542 
543  CKIRNode *nodes = (CKIRNode *)calloc((size_t)count, sizeof(CKIRNode));
544  if (!nodes) {
545  free(buf);
546  return -1;
547  }
548 
549  // Parse each node sequentially.
550  p = nodes_begin;
551  for (int i = 0; i < count; ++i) {
552  // layer
553  char *pl = strstr(p, "\"layer\"");
554  if (!pl) { free(nodes); free(buf); return -1; }
555  int layer = 0;
556  if (sscanf(strchr(pl, ':'), " : %d", &layer) != 1) {
557  free(nodes); free(buf); return -1;
558  }
559 
560  // node
561  char *pn = strstr(pl, "\"node\"");
