17 // Fast exp approximation using polynomial (avoids SVML dependency)
18 // Based on Schraudolph's algorithm with refinement
19 static inline __m512 exp_approx_avx512(__m512 x)
20 {
21  // Clamp x to avoid overflow/underflow
22  const __m512 max_val = _mm512_set1_ps(88.0f);
23  const __m512 min_val = _mm512_set1_ps(-88.0f);
24  x = _mm512_max_ps(_mm512_min_ps(x, max_val), min_val);
25 
26  // exp(x) = 2^(x * log2(e)) = 2^(x * 1.4426950408889634)
27  const __m512 log2e = _mm512_set1_ps(1.4426950408889634f);
28  __m512 z = _mm512_mul_ps(x, log2e);
29 
30  // Split into integer and fractional parts
31  __m512 zf = _mm512_roundscale_ps(z, _MM_FROUND_TO_NEAREST_INT);
32  __m512 f = _mm512_sub_ps(z, zf); // fractional part in [-0.5, 0.5]
33 
34  // Polynomial approximation for 2^f where f in [-0.5, 0.5]
35  // 2^f ≈ 1 + f*ln(2) + f^2*ln(2)^2/2 + f^3*ln(2)^3/6 + ...
36  // Using optimized coefficients
37  const __m512 c0 = _mm512_set1_ps(1.0f);
38  const __m512 c1 = _mm512_set1_ps(0.6931471805599453f); // ln(2)
39  const __m512 c2 = _mm512_set1_ps(0.2402265069591007f); // ln(2)^2/2
40  const __m512 c3 = _mm512_set1_ps(0.05550410866482158f); // ln(2)^3/6
41  const __m512 c4 = _mm512_set1_ps(0.009618129107628478f); // ln(2)^4/24
42 
43  // Horner's method: c0 + f*(c1 + f*(c2 + f*(c3 + f*c4)))
44  __m512 poly = _mm512_fmadd_ps(f, c4, c3);
45  poly = _mm512_fmadd_ps(f, poly, c2);
46  poly = _mm512_fmadd_ps(f, poly, c1);
47  poly = _mm512_fmadd_ps(f, poly, c0);
48 
49  // Scale by 2^n using integer manipulation
50  __m512i zi = _mm512_cvtps_epi32(zf);
51  zi = _mm512_add_epi32(zi, _mm512_set1_epi32(127)); // Add IEEE754 exponent bias
52  zi = _mm512_slli_epi32(zi, 23); // Shift to exponent position
53  __m512 scale = _mm512_castsi512_ps(zi);

56 }
