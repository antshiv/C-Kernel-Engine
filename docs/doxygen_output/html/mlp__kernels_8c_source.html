7 // Forward MLP kernel (FC1 -> GELU -> FC2) adapted from C-Transformer's
8 // mlp_token_parallel but expressed in a model-agnostic form. We keep the
9 // familiar name `mlp_token_parallel` for reuse during decode/inference.
10 //
11 // Shapes:
12 // input: [T × D] (row-major, stride = aligned_dim)
13 // W_fc1: [4D × D] (row-major, stored as [out × in])
14 // b_fc1: [4D]
15 // W_fc2: [D × 4D]
16 // b_fc2: [D]
17 // fc1_output: [T × 4D] (workspace, also becomes GELU input/output)
18 // output: [T × D]

54 // Generic FC2 backward kernel adapted from C-Transformer's backward_fc2_feature_parallel.
55 // Shapes:
56 // d_output: [T × aligned_out]
57 // fc2_input: [T × aligned_in]
58 // W_fc2: [aligned_out × aligned_in] (row-major)
59 // d_input: [T × aligned_in]
60 // d_W_fc2: [aligned_out × aligned_in] (accumulated)
61 // d_b_fc2: [aligned_out] (accumulated)
62 void fc2_backward_kernel(const float *d_output,
63  const float *fc2_input,
64  const float *W_fc2,
65  float *d_input,
66  float *d_W_fc2,
67  float *d_b_fc2,
68  int T,
69  int aligned_in,
70  int aligned_out,
71  int num_threads)
72 {
73  // 1. d_input = d_output @ W_fc2^T
74 #pragma omp parallel for num_threads(num_threads)
75  for (int t = 0; t < T; ++t) {
76  const float *d_out_row = d_output + (size_t)t * aligned_out;
77  float *d_in_row = d_input + (size_t)t * aligned_in;
78 
79  for (int in_idx = 0; in_idx < aligned_in; ++in_idx) {
80  float sum = 0.0f;
81  for (int out_idx = 0; out_idx < aligned_out; ++out_idx) {
82  sum += d_out_row[out_idx] * W_fc2[out_idx * aligned_in + in_idx];
83  }
84  d_in_row[in_idx] = sum;
85  }
86  }
87 
88  // 2. d_W_fc2 = d_output^T @ fc2_input (feature-parallel, vectorized)
89 #pragma omp parallel for schedule(dynamic, 1) num_threads(num_threads)
90  for (int out_idx = 0; out_idx < aligned_out; ++out_idx) {
91  float *dst_row = d_W_fc2 + (size_t)out_idx * aligned_in;
92 
93 #if defined(__AVX512F__)
94  for (int in_idx = 0; in_idx < aligned_in; in_idx += 16) {
95  __m512 accum = _mm512_setzero_ps();
96 
97  for (int t = 0; t < T; ++t) {

99  __m512 grad_broadcast =
100  _mm512_set1_ps(d_output[(size_t)t * aligned_out + out_idx]);
101  accum = _mm512_fmadd_ps(grad_broadcast, input_vec, accum);
102  }
103

105  _mm512_store_ps(dst_row + in_idx, _mm512_add_ps(prev, accum));
106  }

123  _mm256_store_ps(dst_row + in_idx, _mm256_add_ps(prev, accum));
124  }

131  }
