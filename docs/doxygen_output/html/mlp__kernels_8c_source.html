8 // Forward MLP kernel (FC1 -> GELU -> FC2) adapted from C-Transformer's
9 // mlp_token_parallel but expressed in a model-agnostic form. We keep the
10 // familiar name `mlp_token_parallel` for reuse during decode/inference.
11 //
12 // Shapes:
13 // input: [T × D] (row-major, stride = aligned_dim)
14 // W_fc1: [4D × D] (row-major, stored as [out × in])
15 // b_fc1: [4D]
16 // W_fc2: [D × 4D]
17 // b_fc2: [D]
18 // fc1_output: [T × 4D] (workspace, also becomes GELU input/output)
19 // output: [T × D]
20 //
21 // D is typically `aligned_dim` in your transformer; pass that value here.

90 // Generic FC2 backward kernel adapted from C-Transformer's backward_fc2_feature_parallel.
91 // Now uses shared GEMM kernels for d_input and d_W computation.
92 // Shapes:
93 // d_output: [T × aligned_out]
94 // fc2_input: [T × aligned_in]
95 // W_fc2: [aligned_out × aligned_in] (row-major)
96 // d_input: [T × aligned_in]
97 // d_W_fc2: [aligned_out × aligned_in] (accumulated)
98 // d_b_fc2: [aligned_out] (accumulated)
99 void fc2_backward_kernel(const float *d_output,
100  const float *fc2_input,
101  const float *W_fc2,
102  float *d_input,
103  float *d_W_fc2,
104  float *d_b_fc2,
105  int T,
106  int aligned_in,
107  int aligned_out,
108  int num_threads)
109 {
110  (void)num_threads; // Threading handled by GEMM kernels
111 
112  // 1. d_input[T, in] = d_output[T, out] @ W[out, in]
113  // Using gemm_nn: C[M,N] = A[M,K] @ B[K,N]
114  // A = d_output [T, out], B = W [out, in], C = d_input [T, in]
115  // M = T, N = aligned_in, K = aligned_out
116  gemm_nn_avx512(d_output, W_fc2, NULL, d_input,
117  T, aligned_in, aligned_out);
118 
119  // 2. d_W[out, in] = d_output[T, out].T @ fc2_input[T, in]
120  // Using gemm_tn: C[M,N] = A[K,M].T @ B[K,N]
121  // A = d_output [T, out] (stored as [K=T, M=out]), B = fc2_input [T, in]
122  // C = d_W [out, in], M = aligned_out, N = aligned_in, K = T
123  // Note: gemm_tn overwrites, so we need to save and add if accumulating
124  // For now, assume d_W starts zeroed (gradient accumulation handled at higher level)
125  gemm_tn_avx512(d_output, fc2_input, NULL, d_W_fc2,
126  aligned_out, aligned_in, T);
127 
128  // 3. d_b_fc2 = sum_over_T(d_output)
129 #pragma omp parallel for schedule(static)
130  for (int out_idx = 0; out_idx < aligned_out; ++out_idx) {
131  float bias_grad = 0.0f;
132  for (int t = 0; t < T; ++t) {
133  bias_grad += d_output[(size_t)t * aligned_out + out_idx];
134  }
135  d_b_fc2[out_idx] += bias_grad;
136  }
137 }
