1 /**
2  * GEMM Microkernel - High-Performance Register-Blocked Matrix Multiplication
3  *
4  * This file implements optimized GEMM microkernels inspired by oneDNN/BLIS.
5  * The key insight: keep all accumulator values in registers across the K loop.
6  *
7  * Architecture:
8  * 1. Microkernel: Fixed-size tile (8x8) computed entirely in registers
9  * 2. Cache blocking: Outer loops tile for L1/L2/L3 cache
10  * 3. Packing: Optional A/B packing for better memory access patterns
11  *
12  * Supported configurations:
13  * - AVX1 (256-bit, no FMA): 8x8 microkernel using mul+add
14  * - AVX2 (256-bit, FMA): 8x8 microkernel using FMA
15  * - AVX-512 (512-bit, FMA): 8x16 or 16x16 microkernel
16  *
17  * Layout: C[M,N] = A[M,K] @ B[K,N] (row-major)
18  * For transposed B: C[M,N] = A[M,K] @ B[N,K].T
19  */

46 // =============================================================================
47 // 8x8 Microkernel - AVX1 (no FMA)
48 //
49 // Computes: C[0:8, 0:8] += A[0:8, 0:K] @ B[0:K, 0:8]
50 //
51 // Register usage (16 YMM registers available):
52 // - c0-c7: 8 accumulators (one per row of C)
53 // - b: loaded B values
54 // - a: broadcast A values
55 //
56 // This keeps 8 rows x 8 cols = 64 floats in registers!
57 // =============================================================================
58 
59 #if defined(__AVX__)
60 static inline void gemm_microkernel_8x8_avx(
61  int K,
62  const float *A, int lda, // A is [MR, K], lda = stride between rows
63  const float *B, int ldb, // B is [K, NR], ldb = stride between rows
64  float *C, int ldc, // C is [MR, NR], ldc = stride between rows
65  int first_k // If true, zero C; else accumulate
66 )
67 {
68  // Initialize accumulators - one __m256 per row of C
69  __m256 c0, c1, c2, c3, c4, c5, c6, c7;
70 
71  if (first_k) {
72  c0 = _mm256_setzero_ps();
73  c1 = _mm256_setzero_ps();
74  c2 = _mm256_setzero_ps();
75  c3 = _mm256_setzero_ps();
76  c4 = _mm256_setzero_ps();
77  c5 = _mm256_setzero_ps();
78  c6 = _mm256_setzero_ps();
79  c7 = _mm256_setzero_ps();
80  } else {
81  // Load existing C values
82  c0 = _mm256_loadu_ps(&C[0 * ldc]);
83  c1 = _mm256_loadu_ps(&C[1 * ldc]);
84  c2 = _mm256_loadu_ps(&C[2 * ldc]);
85  c3 = _mm256_loadu_ps(&C[3 * ldc]);
86  c4 = _mm256_loadu_ps(&C[4 * ldc]);
87  c5 = _mm256_loadu_ps(&C[5 * ldc]);
88  c6 = _mm256_loadu_ps(&C[6 * ldc]);
89  c7 = _mm256_loadu_ps(&C[7 * ldc]);
90  }
91 
92  // Main K loop - this is where the magic happens
93  // All 64 accumulator values stay in registers!
94  for (int k = 0; k < K; k++) {
95  // Load one row of B: B[k, 0:8]
96  __m256 b = _mm256_loadu_ps(&B[k * ldb]);
97 
98  // For each row of A, broadcast A[row, k] and multiply-accumulate
99  // AVX1: use mul + add (no FMA)
100  __m256 a0 = _mm256_set1_ps(A[0 * lda + k]);
101  __m256 a1 = _mm256_set1_ps(A[1 * lda + k]);
102  __m256 a2 = _mm256_set1_ps(A[2 * lda + k]);
103  __m256 a3 = _mm256_set1_ps(A[3 * lda + k]);
104  __m256 a4 = _mm256_set1_ps(A[4 * lda + k]);
105  __m256 a5 = _mm256_set1_ps(A[5 * lda + k]);
106  __m256 a6 = _mm256_set1_ps(A[6 * lda + k]);
107  __m256 a7 = _mm256_set1_ps(A[7 * lda + k]);
108 
109  c0 = _mm256_add_ps(c0, _mm256_mul_ps(a0, b));
110  c1 = _mm256_add_ps(c1, _mm256_mul_ps(a1, b));
111  c2 = _mm256_add_ps(c2, _mm256_mul_ps(a2, b));
112  c3 = _mm256_add_ps(c3, _mm256_mul_ps(a3, b));
113  c4 = _mm256_add_ps(c4, _mm256_mul_ps(a4, b));
114  c5 = _mm256_add_ps(c5, _mm256_mul_ps(a5, b));
115  c6 = _mm256_add_ps(c6, _mm256_mul_ps(a6, b));
116  c7 = _mm256_add_ps(c7, _mm256_mul_ps(a7, b));
117  }
118 
119  // Store results back to C
120  _mm256_storeu_ps(&C[0 * ldc], c0);
121  _mm256_storeu_ps(&C[1 * ldc], c1);
122  _mm256_storeu_ps(&C[2 * ldc], c2);
123  _mm256_storeu_ps(&C[3 * ldc], c3);
124  _mm256_storeu_ps(&C[4 * ldc], c4);
125  _mm256_storeu_ps(&C[5 * ldc], c5);
126  _mm256_storeu_ps(&C[6 * ldc], c6);
127  _mm256_storeu_ps(&C[7 * ldc], c7);
128 }
129 #endif

259 static void gemm_microkernel_edge(
260  int m, int n, int K,
261  const float *A, int lda,
262  const float *B, int ldb,
263  float *C, int ldc,
264  int first_k
265 )
266 {
267  // Simple scalar fallback for edge tiles
268  for (int i = 0; i < m; i++) {
269  for (int j = 0; j < n; j++) {
270  float sum = first_k ? 0.0f : C[i * ldc + j];

272  sum += A[i * lda + k] * B[k * ldb + j];
273  }
274  C[i * ldc + j] = sum;

279 // =============================================================================
280 // Matrix Packing Functions
281 //
282 // Packing transforms matrices into contiguous layouts optimized for the
283 // microkernel access pattern. This is critical for large matrix performance.
284 // =============================================================================
285 
286 // Pack A panel: A[m0:m0+mc, k0:k0+kc] -> Ap[mc, kc] in row-panel format
287 // Packed layout: for each MR-row panel, store rows contiguously
288 static void pack_a_panel(
289  const float *A, int lda,
290  float *Ap,
291  int mc, int kc, int mr
292 )
293 {
294  for (int i = 0; i < mc; i += mr) {
295  int rows = (i + mr <= mc) ? mr : (mc - i);
296  for (int p = 0; p < rows; p++) {
297  for (int k = 0; k < kc; k++) {
298  Ap[(i / mr) * mr * kc + p * kc + k] = A[(i + p) * lda + k];
299  }
300  }
301  // Zero pad if partial panel
302  for (int p = rows; p < mr; p++) {
303  for (int k = 0; k < kc; k++) {
304  Ap[(i / mr) * mr * kc + p * kc + k] = 0.0f;
305  }
306  }
307  }
308 }
309 
310 // Pack B panel: B[k0:k0+kc, n0:n0+nc] -> Bp[kc, nc] in column-panel format
311 // Packed layout: for each NR-column panel, store columns contiguously by K
312 static void pack_b_panel(
313  const float *B, int ldb,
314  float *Bp,
315  int kc, int nc, int nr
316 )
317 {
318  for (int j = 0; j < nc; j += nr) {
319  int cols = (j + nr <= nc) ? nr : (nc - j);
320  for (int k = 0; k < kc; k++) {
321  for (int q = 0; q < cols; q++) {
322  Bp[(j / nr) * nr * kc + k * nr + q] = B[k * ldb + (j + q)];
323  }
324  // Zero pad if partial panel
325  for (int q = cols; q < nr; q++) {
326  Bp[(j / nr) * nr * kc + k * nr + q] = 0.0f;
327  }
328  }
329  }
330 }

489 void gemm_microkernel_packed(
490  const float *A,
491  const float *B,
492  float *C,
493  int M, int N, int K
494 )
495 {
496 #if defined(__AVX__)
497  const int mr = MR;
498  const int nr = NR;
499 
500  // Allocate packing buffers (thread-local for OpenMP)
501  // Ap: MC x KC, Bp: KC x NC
502  float *Ap = NULL;
503  float *Bp = NULL;
504 
505  // For simplicity, allocate max size buffers
506  // In production, use thread-local storage
507  Ap = (float*)aligned_alloc(32, MC * KC * sizeof(float));
508  Bp = (float*)aligned_alloc(32, KC * NC * sizeof(float));
509 
510  if (!Ap || !Bp) {
511  // Fallback to non-packed version
512  if (Ap) free(Ap);
513  if (Bp) free(Bp);
514  gemm_microkernel_blocked(A, B, C, M, N, K);
515  return;
516  }
517 
518  // Zero C once at the start
519  memset(C, 0, (size_t)M * N * sizeof(float));
520 
521  // Block over K (outermost for B panel reuse)
522  for (int k0 = 0; k0 < K; k0 += KC) {
523  int kb = (k0 + KC <= K) ? KC : (K - k0);
524  int first_k = (k0 == 0);
525 
526  // Block over N
527  for (int n0 = 0; n0 < N; n0 += NC) {
528  int nb = (n0 + NC <= N) ? NC : (N - n0);
529 
530  // Pack B panel: B[k0:k0+kb, n0:n0+nb]
531  pack_b_panel(&B[k0 * N + n0], N, Bp, kb, nb, nr);
532 
533  // Block over M (parallelize this loop)
534  #pragma omp parallel for schedule(static) firstprivate(Ap)
535  for (int m0 = 0; m0 < M; m0 += MC) {
536  int mb = (m0 + MC <= M) ? MC : (M - m0);
537 
538  // Each thread needs its own Ap buffer
539  float *Ap_local = (float*)aligned_alloc(32, MC * KC * sizeof(float));
540  if (!Ap_local) continue;
541 
542  // Pack A panel: A[m0:m0+mb, k0:k0+kb]
543  pack_a_panel(&A[m0 * K + k0], K, Ap_local, mb, kb, mr);
544 
545  // Micro-kernel loop
546  for (int m1 = 0; m1 < mb; m1 += mr) {
547  int mr_actual = (m1 + mr <= mb) ? mr : (mb - m1);
548 
549  for (int n1 = 0; n1 < nb; n1 += nr) {
550  int nr_actual = (n1 + nr <= nb) ? nr : (nb - n1);
551 
552  float *C_tile = &C[(m0 + m1) * N + (n0 + n1)];
553 
554  // Packed A: offset by panel index
555  const float *Ap_tile = &Ap_local[(m1 / mr) * mr * kb];
556  // Packed B: offset by panel index
557  const float *Bp_tile = &Bp[(n1 / nr) * nr * kb];
558 
559  if (mr_actual == mr && nr_actual == nr) {
560  gemm_microkernel_8x8_packed_avx(kb, Ap_tile, Bp_tile, C_tile, N, first_k);
561  } else {
562  // Edge case - use scalar
563  gemm_microkernel_edge(mr_actual, nr_actual, kb,
564  &A[(m0 + m1) * K + k0], K,
565  &B[k0 * N + (n0 + n1)], N,
566  C_tile, N, first_k);
567  }
568  }
569  }
570 
571  free(Ap_local);
572  }
573  }
574  }
575 
576  free(Ap);
577  free(Bp);
578 #else
579  // Fallback
580  gemm_microkernel_blocked(A, B, C, M, N, K);
581 #endif
582 }

584 // =============================================================================
585 // Cache-Blocked GEMM using Microkernels (original, no packing)
586 //
587 // C[M,N] = A[M,K] @ B[K,N]
588 //
589 // The blocking strategy:
590 // 1. Block K into KC chunks (fits A block in L1)
591 // 2. Block N into NC chunks (fits B block in L2)
592 // 3. Block M into MC chunks (fits C block in L2)
593 // 4. Call microkernel for each MRxNR tile
594 // =============================================================================
595 
596 void gemm_microkernel_blocked(
597  const float *A,
598  const float *B,
599  float *C,
600  int M, int N, int K
601 )
602 {
603  // Zero output first
604  memset(C, 0, (size_t)M * N * sizeof(float));
605 
606 #if defined(__AVX512F__)
607  const int nr = 16; // AVX-512: 16 floats per register
608 #else
609  const int nr = NR; // AVX/AVX2: 8 floats per register
610 #endif
611  const int mr = MR;
612 
613  // Block over K (for L1 cache - A panel)
614  for (int k0 = 0; k0 < K; k0 += KC) {
615  int kb = (k0 + KC <= K) ? KC : (K - k0);
616  int first_k = (k0 == 0);
617 
618  // Block over N (for L2/L3 cache - B panel)
619  #pragma omp parallel for schedule(dynamic)
620  for (int n0 = 0; n0 < N; n0 += NC) {
621  int nb = (n0 + NC <= N) ? NC : (N - n0);
622 
623  // Block over M (for L2 cache - A panel reuse)
624  for (int m0 = 0; m0 < M; m0 += MC) {
625  int mb = (m0 + MC <= M) ? MC : (M - m0);
626 
627  // Call microkernel for each MRxNR tile
628  for (int m1 = 0; m1 < mb; m1 += mr) {
629  int mr_actual = (m1 + mr <= mb) ? mr : (mb - m1);
630 
631  for (int n1 = 0; n1 < nb; n1 += nr) {
632  int nr_actual = (n1 + nr <= nb) ? nr : (nb - n1);
633 
634  const float *A_tile = &A[(m0 + m1) * K + k0];
635  const float *B_tile = &B[k0 * N + (n0 + n1)];
636  float *C_tile = &C[(m0 + m1) * N + (n0 + n1)];
637 
638  // Use optimized microkernel for full tiles
639  if (mr_actual == mr && nr_actual == nr) {
640 #if defined(__AVX512F__)
641  gemm_microkernel_8x16_avx512(kb, A_tile, K, B_tile, N, C_tile, N, first_k);
642 #elif defined(__AVX2__) && defined(__FMA__)
643  gemm_microkernel_8x8_fma(kb, A_tile, K, B_tile, N, C_tile, N, first_k);
644 #elif defined(__AVX__)
645  gemm_microkernel_8x8_avx(kb, A_tile, K, B_tile, N, C_tile, N, first_k);
646 #else
647  gemm_microkernel_edge(mr_actual, nr_actual, kb, A_tile, K, B_tile, N, C_tile, N, first_k);
648 #endif
649  } else {
650  // Edge tiles: use scalar fallback
651  gemm_microkernel_edge(mr_actual, nr_actual, kb, A_tile, K, B_tile, N, C_tile, N, first_k);
652  }
653  }
654  }
655  }
656  }
657  }

661 // GEMM with B transposed: C[M,N] = A[M,K] @ B[N,K].T
662 //
663 // This is the common layout for neural network weights (stored as [out, in])

666 #if defined(__AVX__)
667 static inline void gemm_microkernel_8x8_bt_avx(
668  int K,
669  const float *A, int lda, // A is [MR, K], row-major
670  const float *B, int ldb, // B is [NR, K], row-major (transposed)
671  float *C, int ldc,
672  int first_k
673 )
674 {
675  // For B transposed with first_k, zero C first
676  if (first_k) {
677  for (int i = 0; i < 8; i++) {
678  for (int j = 0; j < 8; j++) {
679  C[i * ldc + j] = 0.0f;
680  }
681  }
682  }
683 
684  // For B transposed, we compute dot products differently
685  // C[i,j] = sum_k A[i,k] * B[j,k]
686  //
687  // Strategy: For each k, load A[0:8, k] as 8 broadcasts,
688  // load B[0:8, k] as 8 broadcasts, then compute outer product
689 
690  // Actually, for B transposed we need a different approach
691  // Let's compute 8 dot products at once using horizontal operations
692 
693  // Alternative: Process K in chunks of 8, use SIMD for the K dimension
694  int k = 0;
695  for (; k <= K - 8; k += 8) {
696  // Load 8 consecutive values from each row of A
697  __m256 a0 = _mm256_loadu_ps(&A[0 * lda + k]);
698  __m256 a1 = _mm256_loadu_ps(&A[1 * lda + k]);
699  __m256 a2 = _mm256_loadu_ps(&A[2 * lda + k]);
700  __m256 a3 = _mm256_loadu_ps(&A[3 * lda + k]);
701  __m256 a4 = _mm256_loadu_ps(&A[4 * lda + k]);
702  __m256 a5 = _mm256_loadu_ps(&A[5 * lda + k]);
703  __m256 a6 = _mm256_loadu_ps(&A[6 * lda + k]);
704  __m256 a7 = _mm256_loadu_ps(&A[7 * lda + k]);
705 
706  // For each column j of C (row j of B)
707  for (int j = 0; j < 8; j++) {
708  __m256 b = _mm256_loadu_ps(&B[j * ldb + k]);
709 
710  // Multiply and accumulate partial products
711  // We need horizontal sum at the end
712  __m256 p0 = _mm256_mul_ps(a0, b);
713  __m256 p1 = _mm256_mul_ps(a1, b);
714  __m256 p2 = _mm256_mul_ps(a2, b);
715  __m256 p3 = _mm256_mul_ps(a3, b);
716  __m256 p4 = _mm256_mul_ps(a4, b);
717  __m256 p5 = _mm256_mul_ps(a5, b);
718  __m256 p6 = _mm256_mul_ps(a6, b);
719  __m256 p7 = _mm256_mul_ps(a7, b);
720 
721  // Horizontal sum each product vector
722  // hadd pairs adjacent elements
723  __m256 s01 = _mm256_hadd_ps(p0, p1);
724  __m256 s23 = _mm256_hadd_ps(p2, p3);
725  __m256 s45 = _mm256_hadd_ps(p4, p5);
726  __m256 s67 = _mm256_hadd_ps(p6, p7);
727 
728  __m256 s0123 = _mm256_hadd_ps(s01, s23);
729  __m256 s4567 = _mm256_hadd_ps(s45, s67);
730 
731  // Now we have partial sums, need to combine 128-bit halves
732  __m128 lo0123 = _mm256_castps256_ps128(s0123);
733  __m128 hi0123 = _mm256_extractf128_ps(s0123, 1);
734  __m128 sum0123 = _mm_add_ps(lo0123, hi0123);
735 
736  __m128 lo4567 = _mm256_castps256_ps128(s4567);
737  __m128 hi4567 = _mm256_extractf128_ps(s4567, 1);
738  __m128 sum4567 = _mm_add_ps(lo4567, hi4567);
739 
740  // Extract individual sums and accumulate
741  C[0 * ldc + j] += _mm_cvtss_f32(sum0123);
742  C[1 * ldc + j] += _mm_cvtss_f32(_mm_shuffle_ps(sum0123, sum0123, 1));
743  C[2 * ldc + j] += _mm_cvtss_f32(_mm_shuffle_ps(sum0123, sum0123, 2));
744  C[3 * ldc + j] += _mm_cvtss_f32(_mm_shuffle_ps(sum0123, sum0123, 3));
745  C[4 * ldc + j] += _mm_cvtss_f32(sum4567);
746  C[5 * ldc + j] += _mm_cvtss_f32(_mm_shuffle_ps(sum4567, sum4567, 1));
747  C[6 * ldc + j] += _mm_cvtss_f32(_mm_shuffle_ps(sum4567, sum4567, 2));
748  C[7 * ldc + j] += _mm_cvtss_f32(_mm_shuffle_ps(sum4567, sum4567, 3));
749  }
750  }
751 
752  // Handle remaining K
753  for (; k < K; k++) {
754  for (int i = 0; i < 8; i++) {
755  float a = A[i * lda + k];
756  for (int j = 0; j < 8; j++) {
757  C[i * ldc + j] += a * B[j * ldb + k];
758  }
759  }
760  }
761 }
