103 void gemm_microkernel(
104  const float *A,
105  const float *B,
106  float *C,
107  int M, int N, int K,
108  int B_transposed
109 )
110 {
111  onednn_init();
112 
113  // Create memory descriptors for row-major matrices
114  dnnl_memory_desc_t a_md, b_md, c_md;
115  dnnl_dims_t a_dims = {M, K};
116  dnnl_dims_t b_dims = {K, N};
117  dnnl_dims_t c_dims = {M, N};
118  dnnl_dims_t a_strides = {K, 1};
119  dnnl_dims_t b_strides = {N, 1}; /* default: B is [K,N] row-major */
120  dnnl_dims_t c_strides = {N, 1};
121 
122  if (B_transposed) {
123  /*
124  * Our "B_transposed" convention means: caller stores B as [N,K] row-major,
125  * but wants C = A[M,K] @ B[N,K]^T => treat weights as [K,N].
126  *
127  * oneDNN matmul has no transpose flag, so represent B^T as a strided view:
128  * dims = [K, N]
129  * strides = [1, K] (offset(k,n) = k + n*K == B[n*K + k])
130  */
131  b_strides[0] = 1;
132  b_strides[1] = K;
133  }
134 
135  dnnl_memory_desc_create_with_strides(&a_md, 2, a_dims, dnnl_f32, a_strides);
136  dnnl_memory_desc_create_with_strides(&b_md, 2, b_dims, dnnl_f32, b_strides);
137  dnnl_memory_desc_create_with_strides(&c_md, 2, c_dims, dnnl_f32, c_strides);
138 
139  // Create matmul primitive descriptor
140  dnnl_primitive_desc_t matmul_pd;
141  dnnl_matmul_primitive_desc_create(&matmul_pd, g_engine, a_md, b_md, NULL, c_md, NULL);
142 
143  // Create primitive
144  dnnl_primitive_t matmul;
145  dnnl_primitive_create(&matmul, matmul_pd);
146 
147  // Create memory objects
148  dnnl_memory_t a_mem, b_mem, c_mem;
149  dnnl_memory_create(&a_mem, a_md, g_engine, (void*)A);
150  dnnl_memory_create(&b_mem, b_md, g_engine, (void*)B);
151  dnnl_memory_create(&c_mem, c_md, g_engine, (void*)C);
152 
153  // Execute
154  dnnl_exec_arg_t args[3] = {
155  {DNNL_ARG_SRC, a_mem},
156  {DNNL_ARG_WEIGHTS, b_mem},
157  {DNNL_ARG_DST, c_mem}
158  };
159  dnnl_primitive_execute(matmul, g_stream, 3, args);
160  dnnl_stream_wait(g_stream);
161 
162  // Cleanup
163  dnnl_primitive_destroy(matmul);
164  dnnl_primitive_desc_destroy(matmul_pd);
165  dnnl_memory_destroy(a_mem);
166  dnnl_memory_destroy(b_mem);
167  dnnl_memory_destroy(c_mem);
168  dnnl_memory_desc_destroy(a_md);
169  dnnl_memory_desc_destroy(b_md);
170  dnnl_memory_desc_destroy(c_md);
171 }

222 // =============================================================================
223 // AVX-512 6x32 Microkernel - oneDNN style
224 //
225 // Computes: C[0:6, 0:32] += A[0:6, 0:K] @ B[0:K, 0:32]
226 //
227 // Register usage (32 ZMM registers available):
228 // - c0_lo, c0_hi, c1_lo, c1_hi, ... c5_hi: 12 accumulators (2 per row)
229 // - b_lo, b_hi: 2 registers for B row
230 // - a0-a5: 6 registers for A broadcasts
231 // - Remaining for prefetch and temp
232 // =============================================================================
233 
234 #if defined(__AVX512F__)
235 static inline void gemm_microkernel_6x32_avx512(
236  int K,
237  const float * __restrict__ A, int lda,
238  const float * __restrict__ B, int ldb,
239  float * __restrict__ C, int ldc,
240  int first_k
241 )
242 {
243  // 12 accumulators: 6 rows x 2 ZMM (32 floats) per row
244  __m512 c0_lo, c0_hi, c1_lo, c1_hi, c2_lo, c2_hi;
245  __m512 c3_lo, c3_hi, c4_lo, c4_hi, c5_lo, c5_hi;
246 
247  if (first_k) {
248  c0_lo = _mm512_setzero_ps(); c0_hi = _mm512_setzero_ps();
249  c1_lo = _mm512_setzero_ps(); c1_hi = _mm512_setzero_ps();
250  c2_lo = _mm512_setzero_ps(); c2_hi = _mm512_setzero_ps();
251  c3_lo = _mm512_setzero_ps(); c3_hi = _mm512_setzero_ps();
252  c4_lo = _mm512_setzero_ps(); c4_hi = _mm512_setzero_ps();
253  c5_lo = _mm512_setzero_ps(); c5_hi = _mm512_setzero_ps();
254  } else {
255  c0_lo = _mm512_loadu_ps(&C[0 * ldc]); c0_hi = _mm512_loadu_ps(&C[0 * ldc + 16]);
256  c1_lo = _mm512_loadu_ps(&C[1 * ldc]); c1_hi = _mm512_loadu_ps(&C[1 * ldc + 16]);
257  c2_lo = _mm512_loadu_ps(&C[2 * ldc]); c2_hi = _mm512_loadu_ps(&C[2 * ldc + 16]);
258  c3_lo = _mm512_loadu_ps(&C[3 * ldc]); c3_hi = _mm512_loadu_ps(&C[3 * ldc + 16]);
259  c4_lo = _mm512_loadu_ps(&C[4 * ldc]); c4_hi = _mm512_loadu_ps(&C[4 * ldc + 16]);
260  c5_lo = _mm512_loadu_ps(&C[5 * ldc]); c5_hi = _mm512_loadu_ps(&C[5 * ldc + 16]);
261  }
262 
263  // Prefetch first cache lines
264  _mm_prefetch((const char*)&B[0], _MM_HINT_T0);
265  _mm_prefetch((const char*)&B[64], _MM_HINT_T0);
266 
267  // Main K loop - unrolled by 8 with software pipelining for better ILP
268  int k = 0;
269  for (; k <= K - 8; k += 8) {
270  // Prefetch ahead - 16 rows ahead for L1, 32 for L2
271  _mm_prefetch((const char*)&B[(k + 16) * ldb], _MM_HINT_T0);
272  _mm_prefetch((const char*)&B[(k + 16) * ldb + 64], _MM_HINT_T0);
273  _mm_prefetch((const char*)&B[(k + 32) * ldb], _MM_HINT_T1);
274 
275  // Software pipelining: preload first B row
276  __m512 b_lo_next = _mm512_loadu_ps(&B[k * ldb]);
277  __m512 b_hi_next = _mm512_loadu_ps(&B[k * ldb + 16]);
278 
279  #define AVX512_ITER(koff) { \
280  __m512 b_lo = b_lo_next; \
281  __m512 b_hi = b_hi_next; \
282  if ((koff) < 7) { \
283  b_lo_next = _mm512_loadu_ps(&B[(k + (koff) + 1) * ldb]); \
284  b_hi_next = _mm512_loadu_ps(&B[(k + (koff) + 1) * ldb + 16]); \
285  } \
286  __m512 a0 = _mm512_set1_ps(A[0 * lda + k + (koff)]); \
287  __m512 a1 = _mm512_set1_ps(A[1 * lda + k + (koff)]); \
288  __m512 a2 = _mm512_set1_ps(A[2 * lda + k + (koff)]); \
289  __m512 a3 = _mm512_set1_ps(A[3 * lda + k + (koff)]); \
290  __m512 a4 = _mm512_set1_ps(A[4 * lda + k + (koff)]); \
291  __m512 a5 = _mm512_set1_ps(A[5 * lda + k + (koff)]); \
292  c0_lo = _mm512_fmadd_ps(a0, b_lo, c0_lo); c0_hi = _mm512_fmadd_ps(a0, b_hi, c0_hi); \
293  c1_lo = _mm512_fmadd_ps(a1, b_lo, c1_lo); c1_hi = _mm512_fmadd_ps(a1, b_hi, c1_hi); \
294  c2_lo = _mm512_fmadd_ps(a2, b_lo, c2_lo); c2_hi = _mm512_fmadd_ps(a2, b_hi, c2_hi); \
295  c3_lo = _mm512_fmadd_ps(a3, b_lo, c3_lo); c3_hi = _mm512_fmadd_ps(a3, b_hi, c3_hi); \
296  c4_lo = _mm512_fmadd_ps(a4, b_lo, c4_lo); c4_hi = _mm512_fmadd_ps(a4, b_hi, c4_hi); \
297  c5_lo = _mm512_fmadd_ps(a5, b_lo, c5_lo); c5_hi = _mm512_fmadd_ps(a5, b_hi, c5_hi); \
298  }
299 
300  AVX512_ITER(0);
301  AVX512_ITER(1);
302  AVX512_ITER(2);
303  AVX512_ITER(3);
304  AVX512_ITER(4);
305  AVX512_ITER(5);
306  AVX512_ITER(6);
307  AVX512_ITER(7);
308 
309  #undef AVX512_ITER
310  }
311 
312  // Handle K % 8 remainder with 4-unroll
313  for (; k <= K - 4; k += 4) {
314  #define AVX512_ITER4(koff) { \
315  __m512 b_lo = _mm512_loadu_ps(&B[(k + koff) * ldb]); \
316  __m512 b_hi = _mm512_loadu_ps(&B[(k + koff) * ldb + 16]); \
317  __m512 a0 = _mm512_set1_ps(A[0 * lda + k + koff]); \
318  __m512 a1 = _mm512_set1_ps(A[1 * lda + k + koff]); \
319  __m512 a2 = _mm512_set1_ps(A[2 * lda + k + koff]); \
320  __m512 a3 = _mm512_set1_ps(A[3 * lda + k + koff]); \
321  __m512 a4 = _mm512_set1_ps(A[4 * lda + k + koff]); \
322  __m512 a5 = _mm512_set1_ps(A[5 * lda + k + koff]); \
323  c0_lo = _mm512_fmadd_ps(a0, b_lo, c0_lo); c0_hi = _mm512_fmadd_ps(a0, b_hi, c0_hi); \
324  c1_lo = _mm512_fmadd_ps(a1, b_lo, c1_lo); c1_hi = _mm512_fmadd_ps(a1, b_hi, c1_hi); \
325  c2_lo = _mm512_fmadd_ps(a2, b_lo, c2_lo); c2_hi = _mm512_fmadd_ps(a2, b_hi, c2_hi); \
326  c3_lo = _mm512_fmadd_ps(a3, b_lo, c3_lo); c3_hi = _mm512_fmadd_ps(a3, b_hi, c3_hi); \
327  c4_lo = _mm512_fmadd_ps(a4, b_lo, c4_lo); c4_hi = _mm512_fmadd_ps(a4, b_hi, c4_hi); \
328  c5_lo = _mm512_fmadd_ps(a5, b_lo, c5_lo); c5_hi = _mm512_fmadd_ps(a5, b_hi, c5_hi); \
329  }
330  AVX512_ITER4(0);
331  AVX512_ITER4(1);
332  AVX512_ITER4(2);
333  AVX512_ITER4(3);
334  #undef AVX512_ITER4
335  }
336 
337  // Handle remaining K
338  for (; k < K; k++) {
339  __m512 b_lo = _mm512_loadu_ps(&B[k * ldb]);
340  __m512 b_hi = _mm512_loadu_ps(&B[k * ldb + 16]);
341 
342  c0_lo = _mm512_fmadd_ps(_mm512_set1_ps(A[0 * lda + k]), b_lo, c0_lo);
343  c0_hi = _mm512_fmadd_ps(_mm512_set1_ps(A[0 * lda + k]), b_hi, c0_hi);
344  c1_lo = _mm512_fmadd_ps(_mm512_set1_ps(A[1 * lda + k]), b_lo, c1_lo);
345  c1_hi = _mm512_fmadd_ps(_mm512_set1_ps(A[1 * lda + k]), b_hi, c1_hi);
346  c2_lo = _mm512_fmadd_ps(_mm512_set1_ps(A[2 * lda + k]), b_lo, c2_lo);
347  c2_hi = _mm512_fmadd_ps(_mm512_set1_ps(A[2 * lda + k]), b_hi, c2_hi);
348  c3_lo = _mm512_fmadd_ps(_mm512_set1_ps(A[3 * lda + k]), b_lo, c3_lo);
349  c3_hi = _mm512_fmadd_ps(_mm512_set1_ps(A[3 * lda + k]), b_hi, c3_hi);
350  c4_lo = _mm512_fmadd_ps(_mm512_set1_ps(A[4 * lda + k]), b_lo, c4_lo);
351  c4_hi = _mm512_fmadd_ps(_mm512_set1_ps(A[4 * lda + k]), b_hi, c4_hi);
352  c5_lo = _mm512_fmadd_ps(_mm512_set1_ps(A[5 * lda + k]), b_lo, c5_lo);
353  c5_hi = _mm512_fmadd_ps(_mm512_set1_ps(A[5 * lda + k]), b_hi, c5_hi);
354  }
355 
356  // Store results
357  _mm512_storeu_ps(&C[0 * ldc], c0_lo); _mm512_storeu_ps(&C[0 * ldc + 16], c0_hi);
358  _mm512_storeu_ps(&C[1 * ldc], c1_lo); _mm512_storeu_ps(&C[1 * ldc + 16], c1_hi);
359  _mm512_storeu_ps(&C[2 * ldc], c2_lo); _mm512_storeu_ps(&C[2 * ldc + 16], c2_hi);
360  _mm512_storeu_ps(&C[3 * ldc], c3_lo); _mm512_storeu_ps(&C[3 * ldc + 16], c3_hi);
361  _mm512_storeu_ps(&C[4 * ldc], c4_lo); _mm512_storeu_ps(&C[4 * ldc + 16], c4_hi);
362  _mm512_storeu_ps(&C[5 * ldc], c5_lo); _mm512_storeu_ps(&C[5 * ldc + 16], c5_hi);
363 }

368  const float * __restrict__ Ap, // Packed A: [MR, K] contiguous

370  float * __restrict__ C, int ldc,
371  int first_k
372 )
373 {

375  __m512 c3_lo, c3_hi, c4_lo, c4_hi, c5_lo, c5_hi;
376 
377  if (first_k) {
378  c0_lo = _mm512_setzero_ps(); c0_hi = _mm512_setzero_ps();
379  c1_lo = _mm512_setzero_ps(); c1_hi = _mm512_setzero_ps();
380  c2_lo = _mm512_setzero_ps(); c2_hi = _mm512_setzero_ps();
381  c3_lo = _mm512_setzero_ps(); c3_hi = _mm512_setzero_ps();
382  c4_lo = _mm512_setzero_ps(); c4_hi = _mm512_setzero_ps();
383  c5_lo = _mm512_setzero_ps(); c5_hi = _mm512_setzero_ps();
384  } else {
385  c0_lo = _mm512_loadu_ps(&C[0 * ldc]); c0_hi = _mm512_loadu_ps(&C[0 * ldc + 16]);
386  c1_lo = _mm512_loadu_ps(&C[1 * ldc]); c1_hi = _mm512_loadu_ps(&C[1 * ldc + 16]);
387  c2_lo = _mm512_loadu_ps(&C[2 * ldc]); c2_hi = _mm512_loadu_ps(&C[2 * ldc + 16]);
388  c3_lo = _mm512_loadu_ps(&C[3 * ldc]); c3_hi = _mm512_loadu_ps(&C[3 * ldc + 16]);
389  c4_lo = _mm512_loadu_ps(&C[4 * ldc]); c4_hi = _mm512_loadu_ps(&C[4 * ldc + 16]);
390  c5_lo = _mm512_loadu_ps(&C[5 * ldc]); c5_hi = _mm512_loadu_ps(&C[5 * ldc + 16]);
391  }
392 
393  // Packed B is contiguous: B[k, 0:32] at Bp[k * 32]
394  _mm_prefetch((const char*)Bp, _MM_HINT_T0);
395  _mm_prefetch((const char*)(Bp + 16), _MM_HINT_T0);
396 
397  int k = 0;
398  for (; k <= K - 4; k += 4) {
399  _mm_prefetch((const char*)(Bp + (k + 8) * NR), _MM_HINT_T0);
400  _mm_prefetch((const char*)(Bp + (k + 8) * NR + 16), _MM_HINT_T0);
401 
402  #define PACKED_ITER(koff) { \
403  __m512 b_lo = _mm512_load_ps(&Bp[(k + koff) * NR]); \
404  __m512 b_hi = _mm512_load_ps(&Bp[(k + koff) * NR + 16]); \
405  __m512 a0 = _mm512_set1_ps(Ap[0 * K + k + koff]); \
406  __m512 a1 = _mm512_set1_ps(Ap[1 * K + k + koff]); \
407  __m512 a2 = _mm512_set1_ps(Ap[2 * K + k + koff]); \
408  __m512 a3 = _mm512_set1_ps(Ap[3 * K + k + koff]); \
409  __m512 a4 = _mm512_set1_ps(Ap[4 * K + k + koff]); \
410  __m512 a5 = _mm512_set1_ps(Ap[5 * K + k + koff]); \
411  c0_lo = _mm512_fmadd_ps(a0, b_lo, c0_lo); c0_hi = _mm512_fmadd_ps(a0, b_hi, c0_hi); \
412  c1_lo = _mm512_fmadd_ps(a1, b_lo, c1_lo); c1_hi = _mm512_fmadd_ps(a1, b_hi, c1_hi); \

750 // Pack A panel: A[m0:m0+mc, k0:k0+kc] -> Ap[mc, kc] in row-panel format
