26 static inline int ck_nearest_int(float fval) {
27  /* Bit-level round-to-nearest from llama.cpp (fast + deterministic). */
28  float val = fval + 12582912.f;
29  int i;
30  memcpy(&i, &val, sizeof(int));
31  return (i & 0x007fffff) - 0x00400000;
32 }

34 void quantize_row_q8_k(const float *x, void *vy, int k) {
35  if (!x || !vy || k <= 0) {
36  return;
37  }
38  assert(k % QK_K == 0);
39  const int nb = k / QK_K;
40  block_q8_K *y = (block_q8_K *)vy;
41 
42  for (int i = 0; i < nb; ++i) {
43  float max = 0.0f;
44  float amax = 0.0f;
45  for (int j = 0; j < QK_K; ++j) {
46  float ax = fabsf(x[j]);
47  if (ax > amax) {
48  amax = ax;
49  max = x[j];
50  }
51  }
52  if (!amax) {
53  y[i].d = 0.0f;
54  memset(y[i].qs, 0, sizeof(y[i].qs));
55  memset(y[i].bsums, 0, sizeof(y[i].bsums));
56  x += QK_K;
57  continue;
58  }
59 
60  const float iscale = -127.0f / max;
61  for (int j = 0; j < QK_K; ++j) {
62  int v = ck_nearest_int(iscale * x[j]);
63  if (v > 127) {
64  v = 127;
65  }
66  if (v < -128) {
67  v = -128;
68  }
69  y[i].qs[j] = (int8_t)v;
70  }
71 
72  for (int j = 0; j < QK_K / 16; ++j) {
73  int sum = 0;
74  const int8_t *qs = &y[i].qs[j * 16];
75  for (int ii = 0; ii < 16; ++ii) {
76  sum += qs[ii];
77  }
78  y[i].bsums[j] = (int16_t)sum;
79  }
80 
81  y[i].d = 1.0f / iscale;
82  x += QK_K;
83  }
84 }

86 static float dot_q4_k_q8_k_ref(const block_q4_K *w,
87  const block_q8_K *x,
88  int k)
89 {
90  const int nb = k / QK_K;
91  float sumf = 0.0f;
92 
93  for (int i = 0; i < nb; ++i) {
94  uint8_t sc[8], m[8];
95  unpack_q4_k_scales(w[i].scales, sc, m);
96 
97  int32_t sum_scale = 0;
98  int32_t sum_min = 0;
99 
100  for (int sub = 0; sub < 8; ++sub) {
101  const uint8_t *qs = &w[i].qs[sub * 16];
102  const int8_t *q8 = &x[i].qs[sub * 32];
103 
104  int32_t sum_q4q8 = 0;
105  for (int j = 0; j < 16; ++j) {
106  const uint8_t packed = qs[j];
107  const int q4_0 = (int)(packed & 0x0F) - 8;
108  const int q4_1 = (int)(packed >> 4) - 8;
109  sum_q4q8 += q4_0 * q8[2 * j];
110  sum_q4q8 += q4_1 * q8[2 * j + 1];
111  }
112 
113  const int32_t sum_q8 = (int32_t)x[i].bsums[sub * 2] +
114  (int32_t)x[i].bsums[sub * 2 + 1];
115  sum_scale += (int32_t)sc[sub] * sum_q4q8;
116  sum_min += (int32_t)m[sub] * sum_q8;
117  }
118 
119  const float d = CK_FP16_TO_FP32(w[i].d) * x[i].d;
120  const float dmin = CK_FP16_TO_FP32(w[i].dmin) * x[i].d;
121  sumf += d * (float)sum_scale;
122  sumf += dmin * (float)sum_min;
123  }
124 
125  return sumf;
126 }
