1 /**
2  * Fused GEMM Kernels
3  *
4  * GEMM operations fused with activations (ReLU, GELU, SiLU) and SwiGLU.
5  * The key benefit: intermediate results stay in registers, avoiding DRAM
6  * round-trips between operations.
7  *
8  * Supported operations:
9  * - gemm_bias_relu_fused: C = ReLU(A @ B^T + bias)
10  * - gemm_bias_gelu_fused: C = GELU(A @ B^T + bias)
11  * - gemm_bias_silu_fused: C = SiLU(A @ B^T + bias)
12  * - gemm_swiglu_fused: C = SiLU(x @ W_gate) * (x @ W_up)
13  *
14  * All kernels support:
15  * - AVX1 SIMD (256-bit vectors, no FMA)
16  * - OpenMP parallelization
17  * - Scalar fallback
18  */

35 #if defined(__AVX__) || defined(__AVX512F__)
36 // Horizontal sum of 8 floats in __m256 (works for both AVX and AVX-512)
37 static inline float hsum256_ps_fused(__m256 v) {
38  __m128 lo = _mm256_castps256_ps128(v);
39  __m128 hi = _mm256_extractf128_ps(v, 1);
40  __m128 sum128 = _mm_add_ps(lo, hi);
41  __m128 shuf = _mm_movehdup_ps(sum128);
42  __m128 sums = _mm_add_ps(sum128, shuf);
43  shuf = _mm_movehl_ps(shuf, sums);
44  sums = _mm_add_ss(sums, shuf);
45  return _mm_cvtss_f32(sums);
46 }
47 #endif
48 
49 #if defined(__AVX512F__)
50 // Horizontal sum of 16 floats in __m512
51 static inline float hsum512_ps_fused(__m512 v) {
52  __m256 lo = _mm512_castps512_ps256(v);
53  __m256 hi = _mm512_extractf32x8_ps(v, 1);
54  __m256 sum256 = _mm256_add_ps(lo, hi);
55  return hsum256_ps_fused(sum256);
56 }

63 // GELU approximation: x * sigmoid(1.702 * x) (QuickGELU)
64 static inline float fast_gelu_scalar(float x) {
65  float sx = 1.702f * x;
66  float sig = 1.0f / (1.0f + expf(-sx));
67  return x * sig;
68 }

71 // GEMM + Bias + ReLU fused
72 // C[i,j] = max(0, sum_k(A[i,k] * B[j,k]) + bias[j])
73 // =============================================================================
74 void gemm_bias_relu_fused(const float *A,
75  const float *B,
76  const float *bias,
77  float *C,
78  int M, int N, int K)
79 {
80 #if defined(__AVX__)
81 #pragma omp parallel for
82  for (int i = 0; i < M; i++) {
83  for (int j = 0; j < N; j++) {
84  __m256 sum_vec = _mm256_setzero_ps();
85  int k;
86  for (k = 0; k <= K - 8; k += 8) {
87  __m256 a_vec = _mm256_loadu_ps(&A[i * K + k]);
88  __m256 b_vec = _mm256_loadu_ps(&B[j * K + k]);
89  __m256 prod = _mm256_mul_ps(a_vec, b_vec);
90  sum_vec = _mm256_add_ps(sum_vec, prod);
91  }
92  float sum = hsum256_ps_fused(sum_vec);
93  for (; k < K; k++) {
94  sum += A[i * K + k] * B[j * K + k];
95  }
96  // Fused: add bias and ReLU while still in register
97  sum += bias[j];
98  C[i * N + j] = sum > 0.0f ? sum : 0.0f;
99  }
100  }
101 #else
102 #pragma omp parallel for
103  for (int i = 0; i < M; i++) {
104  for (int j = 0; j < N; j++) {
105  float sum = 0.0f;
106  for (int k = 0; k < K; k++) {
107  sum += A[i * K + k] * B[j * K + k];
108  }
109  sum += bias[j];
110  C[i * N + j] = sum > 0.0f ? sum : 0.0f;
111  }
112  }
113 #endif
114 }

211 // =============================================================================
212 // GEMM + SwiGLU Fused (LLaMA/SmolLM style MLP)
213 //
214 // Computes: output = SiLU(x @ W_gate + b_gate) * (x @ W_up + b_up)
215 //
216 // This fuses TWO GEMMs + SwiGLU activation into one pass:
217 // - gate = x @ W_gate + b_gate (GEMM 1)
218 // - up = x @ W_up + b_up (GEMM 2)
219 // - out = SiLU(gate) * up (SwiGLU)
220 //
221 // Layout:
222 // x: [M, K] input activations
223 // W_gate: [N, K] gate projection weights (transposed)
224 // W_up: [N, K] up projection weights (transposed)
225 // b_gate: [N] gate bias (can be NULL)
226 // b_up: [N] up bias (can be NULL)
227 // output: [M, N] result
228 //
229 // The key insight: gate and up values stay in registers, never written to DRAM
230 // =============================================================================
231 void gemm_swiglu_fused(const float *x,
232  const float *W_gate,
233  const float *W_up,
234  const float *b_gate,
235  const float *b_up,
236  float *output,
237  int M, int N, int K)
238 {
239 #if defined(__AVX__)
240 #pragma omp parallel for
241  for (int i = 0; i < M; i++) {
242  const float *x_row = &x[i * K];
243  float *out_row = &output[i * N];
244 
245  for (int j = 0; j < N; j++) {
246  const float *w_gate_row = &W_gate[j * K];
247  const float *w_up_row = &W_up[j * K];
248 
249  // Compute both dot products in parallel using SIMD
250  __m256 gate_vec = _mm256_setzero_ps();
251  __m256 up_vec = _mm256_setzero_ps();
252 
253  int k;
254  for (k = 0; k <= K - 8; k += 8) {
255  __m256 x_vec = _mm256_loadu_ps(&x_row[k]);
256  __m256 wg_vec = _mm256_loadu_ps(&w_gate_row[k]);
257  __m256 wu_vec = _mm256_loadu_ps(&w_up_row[k]);
258 
259  // gate += x * W_gate
260  gate_vec = _mm256_add_ps(gate_vec, _mm256_mul_ps(x_vec, wg_vec));
261  // up += x * W_up
262  up_vec = _mm256_add_ps(up_vec, _mm256_mul_ps(x_vec, wu_vec));
263  }
264 
265  // Horizontal sum
266  float gate = hsum256_ps_fused(gate_vec);
267  float up = hsum256_ps_fused(up_vec);
268 
269  // Scalar remainder
270  for (; k < K; k++) {
271  gate += x_row[k] * w_gate_row[k];
272  up += x_row[k] * w_up_row[k];
273  }
274 
275  // Add biases
276  if (b_gate) gate += b_gate[j];
277  if (b_up) up += b_up[j];
278 
279  // SwiGLU: SiLU(gate) * up = gate * sigmoid(gate) * up
280  float sig = 1.0f / (1.0f + expf(-gate));
281  out_row[j] = gate * sig * up;
282  }
283  }
284 #else
285  // Scalar fallback
286 #pragma omp parallel for
287  for (int i = 0; i < M; i++) {
288  for (int j = 0; j < N; j++) {
289  float gate = 0.0f;
290  float up = 0.0f;
291 
292  for (int k = 0; k < K; k++) {
293  gate += x[i * K + k] * W_gate[j * K + k];
294  up += x[i * K + k] * W_up[j * K + k];
295  }
296 
297  if (b_gate) gate += b_gate[j];
298  if (b_up) up += b_up[j];
299 
300  // SwiGLU: SiLU(gate) * up
301  float sig = 1.0f / (1.0f + expf(-gate));
302  output[i * N + j] = gate * sig * up;
303  }
304  }
305 #endif
306 }
