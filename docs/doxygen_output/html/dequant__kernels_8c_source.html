3  * @brief Dequantization kernels for GGML-compatible formats

5  * Implements dequantization from Q4_0, Q4_K, Q8_0 to FP32.

23 /**
24  * @brief Dequantize a single Q4_0 block to FP32
25  * @param block Pointer to Q4_0 block (18 bytes)
26  * @param output Output FP32 array (32 floats)
27  */
28 void dequant_q4_0_block(const block_q4_0 *block, float *output)
29 {
30  const float d = GGML_FP16_TO_FP32(block->d);
31 
32  for (int i = 0; i < QK4_0 / 2; i++) {
33  const uint8_t packed = block->qs[i];
34 
35  /* Lower nibble: subtract 8 to get signed value */
36  const int8_t q0 = (packed & 0x0F) - 8;
37  /* Upper nibble: subtract 8 to get signed value */
38  const int8_t q1 = (packed >> 4) - 8;
39 
40  output[2*i + 0] = d * (float)q0;
41  output[2*i + 1] = d * (float)q1;
42  }
43 }

62 /**
63  * @brief Dequantize Q4_0 block using AVX-512 (16 floats at a time)
64  * @param block Pointer to Q4_0 block
65  * @param out_lo Lower 16 floats (weights 0-15)
66  * @param out_hi Upper 16 floats (weights 16-31)
67  */
68 void dequant_q4_0_block_avx512(const block_q4_0 *block,
69  __m512 *out_lo, __m512 *out_hi)
70 {
71  const __m512 scale = _mm512_set1_ps(GGML_FP16_TO_FP32(block->d));
72  const __m512i offset = _mm512_set1_epi32(8);
73 
74  /* Load 16 bytes = 32 x 4-bit weights */
75  __m128i packed = _mm_loadu_si128((const __m128i *)block->qs);
76 
77  /* Unpack lower nibbles (weights 0, 2, 4, ...) */
78  __m512i lo_nibbles = _mm512_cvtepu8_epi32(packed);
79  lo_nibbles = _mm512_and_epi32(lo_nibbles, _mm512_set1_epi32(0x0F));
80  lo_nibbles = _mm512_sub_epi32(lo_nibbles, offset);
81 
82  /* Unpack upper nibbles (weights 1, 3, 5, ...) */
83  __m512i hi_nibbles = _mm512_cvtepu8_epi32(packed);
84  hi_nibbles = _mm512_srli_epi32(hi_nibbles, 4);
85  hi_nibbles = _mm512_sub_epi32(hi_nibbles, offset);
86 
87  /* Convert to float and scale */
88  *out_lo = _mm512_mul_ps(_mm512_cvtepi32_ps(lo_nibbles), scale);
89  *out_hi = _mm512_mul_ps(_mm512_cvtepi32_ps(hi_nibbles), scale);
90 
91  /* Note: This gives interleaved output (0,2,4... and 1,3,5...)
92  * For proper sequential order, would need shuffle/blend */
93 }
94 #endif /* __AVX512F__ */

129  * @brief Dequantize Q8_0 block using AVX-512

131 void dequant_q8_0_block_avx512(const block_q8_0 *block,
132  __m512 *out0, __m512 *out1)
133 {
134  const __m512 scale = _mm512_set1_ps(GGML_FP16_TO_FP32(block->d));
135 
136  /* Load 32 x int8 as two __m128i */
137  __m128i q0 = _mm_loadu_si128((const __m128i *)&block->qs[0]);
138  __m128i q1 = _mm_loadu_si128((const __m128i *)&block->qs[16]);
139 
140  /* Sign-extend to 32-bit and convert to float */
141  __m512i i0 = _mm512_cvtepi8_epi32(q0);
142  __m512i i1 = _mm512_cvtepi8_epi32(q1);
143 
144  *out0 = _mm512_mul_ps(_mm512_cvtepi32_ps(i0), scale);
145  *out1 = _mm512_mul_ps(_mm512_cvtepi32_ps(i1), scale);
146 }

149 /* ============================================================================
150  * Q4_K Dequantization (Primary Target for Q4_K_M)
151  * - 256 weights per super-block
152  * - 8 sub-blocks of 32 weights each
153  * - Two-level scaling: super-block d/dmin + sub-block 6-bit scales
154  * ============================================================================ */
155 
156 /**
157  * @brief Dequantize a single Q4_K block to FP32
158  *
159  * Q4_K uses nested scales:
160  * w_fp32 = q * (d * sub_scale) + dmin * sub_min
161  *
162  * Where:
163  * - d, dmin are FP16 super-block values
164  * - sub_scale, sub_min are 6-bit values packed in scales[12]
165  * - q is the 4-bit quantized weight (-8 to +7 after offset)
166  */
167 void dequant_q4_k_block(const block_q4_K *block, float *output)
168 {
169  const float d = GGML_FP16_TO_FP32(block->d);
170  const float dmin = GGML_FP16_TO_FP32(block->dmin);
171 
172  /* Unpack the 6-bit sub-block scales and mins */
173  uint8_t sc[8], m[8];
174  unpack_q4_k_scales(block->scales, sc, m);
175 
176  /* Process 8 sub-blocks of 32 weights each */
177  for (int sub = 0; sub < 8; sub++) {
178  const float scale = d * (float)sc[sub];
179  const float min_val = dmin * (float)m[sub];
180 
181  /* Each sub-block has 32 weights = 16 bytes */
182  const uint8_t *qs = &block->qs[sub * 16];
183  float *out = &output[sub * 32];
184 
185  for (int i = 0; i < 16; i++) {
186  const uint8_t packed = qs[i];
187 
188  /* Lower nibble */
189  const int8_t q0 = (packed & 0x0F) - 8;
190  /* Upper nibble */
191  const int8_t q1 = (packed >> 4) - 8;
192 
193  out[2*i + 0] = scale * (float)q0 + min_val;
194  out[2*i + 1] = scale * (float)q1 + min_val;
195  }
196  }
197 }

212 #ifdef __AVX512F__
213 /**
214  * @brief Dequantize one Q4_K sub-block (32 weights) using AVX-512
215  *
216  * @param qs Pointer to 16 bytes of packed 4-bit weights
217  * @param scale Pre-computed d * sub_scale
218  * @param min_val Pre-computed dmin * sub_min
219  * @param out0 Output: weights 0-15
220  * @param out1 Output: weights 16-31
221  */
222 static inline void dequant_q4_k_subblock_avx512(
223  const uint8_t *qs,
224  float scale,
225  float min_val,
226  __m512 *out0,
227  __m512 *out1)
228 {
229  const __m512 vscale = _mm512_set1_ps(scale);
230  const __m512 vmin = _mm512_set1_ps(min_val);
231  const __m512i offset = _mm512_set1_epi32(8);
232  const __m512i mask_lo = _mm512_set1_epi32(0x0F);
233 
234  /* Load 16 bytes = 32 x 4-bit weights */
235  __m128i packed = _mm_loadu_si128((const __m128i *)qs);
236 
237  /* Expand to 32-bit for lower and upper nibbles */
238  __m512i bytes = _mm512_cvtepu8_epi32(packed);
239 
240  /* Extract lower nibbles */
241  __m512i lo = _mm512_and_epi32(bytes, mask_lo);
242  lo = _mm512_sub_epi32(lo, offset);
243 
244  /* Extract upper nibbles */
245  __m512i hi = _mm512_srli_epi32(bytes, 4);
246  hi = _mm512_sub_epi32(hi, offset);
247 
248  /* Convert to float, scale, and add min */
249  *out0 = _mm512_fmadd_ps(_mm512_cvtepi32_ps(lo), vscale, vmin);
250  *out1 = _mm512_fmadd_ps(_mm512_cvtepi32_ps(hi), vscale, vmin);
251 }

253 /**
254  * @brief Dequantize full Q4_K block using AVX-512
255  * @param block Q4_K block (144 bytes)
256  * @param output 256 floats output
257  */
258 void dequant_q4_k_block_avx512(const block_q4_K *block, float *output)
259 {
260  const float d = GGML_FP16_TO_FP32(block->d);
261  const float dmin = GGML_FP16_TO_FP32(block->dmin);
262 
263  uint8_t sc[8], m[8];
264  unpack_q4_k_scales(block->scales, sc, m);
265 
266  for (int sub = 0; sub < 8; sub++) {
267  const float scale = d * (float)sc[sub];
268  const float min_val = dmin * (float)m[sub];
269 
270  __m512 out0, out1;
271  dequant_q4_k_subblock_avx512(&block->qs[sub * 16], scale, min_val,
272  &out0, &out1);
273 
274  /* Store interleaved - need to de-interleave for correct order */
275  /* For now, store as-is (caller must handle interleaving) */
276  _mm512_storeu_ps(&output[sub * 32], out0);
277  _mm512_storeu_ps(&output[sub * 32 + 16], out1);
278  }
279 }
