2  * @file add_kernels_bf16.c
3  * @brief Element-wise addition kernels for BF16 tensors
4  *
5  * Used for residual connections in transformer models:
6  * residual = x + sublayer_output

8  * Supports:
9  * - Forward: y = a + b
10  * - Forward with scale: y = a + alpha * b
11  * - Backward: d_a = d_y, d_b = d_y (gradient flows through unchanged)

25 /* =============================================================================
26  * Forward: y = a + b
27  * ============================================================================= */
28 
29 void add_forward_bf16(const uint16_t *a,
30  const uint16_t *b,
31  uint16_t *y,
32  size_t n)
33 {
34  if (!a || !b || !y || n == 0) {
35  return;
36  }
37 
38  size_t i = 0;
39 
40 #if defined(__AVX512F__)
41  /* AVX-512: Process 16 bf16 elements at a time */
42  for (; i + 16 <= n; i += 16) {
43  __m512 av = bf16_loadu_cvt_fp32(&a[i]);
44  __m512 bv = bf16_loadu_cvt_fp32(&b[i]);
45  __m512 yv = _mm512_add_ps(av, bv);
46  fp32_cvt_storeu_bf16(&y[i], yv);
47  }
48 #endif
49 
50  /* Scalar fallback */
51  for (; i < n; ++i) {
52  float af = bf16_to_float(a[i]);
53  float bf = bf16_to_float(b[i]);
54  y[i] = float_to_bf16(af + bf);
55  }
56 }

154 /* =============================================================================
155  * Backward: d_a = d_y, d_b = d_y
156  *
157  * For y = a + b, gradients pass through unchanged:
158  * dy/da = 1, dy/db = 1
159  *
160  * This is a simple copy operation, but we provide it for API consistency.
161  * If d_a == d_y or d_b == d_y (in-place), no copy needed.
162  * ============================================================================= */
163 
164 void add_backward_bf16(const uint16_t *d_y,
165  uint16_t *d_a,
166  uint16_t *d_b,
167  size_t n)
168 {
169  if (!d_y || n == 0) {
170  return;
171  }
172 
173  size_t i = 0;
174 
175  /* Copy to d_a if not in-place */
176  if (d_a && d_a != d_y) {
177 #if defined(__AVX512F__)
178  for (; i + 32 <= n; i += 32) {
179  __m512i v0 = _mm512_loadu_si512((const __m512i*)&d_y[i]);
180  __m512i v1 = _mm512_loadu_si512((const __m512i*)&d_y[i + 32]);
181  _mm512_storeu_si512((__m512i*)&d_a[i], v0);
182  _mm512_storeu_si512((__m512i*)&d_a[i + 32], v1);
183  }
184 #endif
185  for (; i < n; ++i) {
186  d_a[i] = d_y[i];
187  }
188  }
189 
190  /* Copy to d_b if not in-place */
191  i = 0;
192  if (d_b && d_b != d_y) {
193 #if defined(__AVX512F__)
194  for (; i + 32 <= n; i += 32) {
195  __m512i v0 = _mm512_loadu_si512((const __m512i*)&d_y[i]);
196  __m512i v1 = _mm512_loadu_si512((const __m512i*)&d_y[i + 32]);
197  _mm512_storeu_si512((__m512i*)&d_b[i], v0);
198  _mm512_storeu_si512((__m512i*)&d_b[i + 32], v1);
199  }
200 #endif
201  for (; i < n; ++i) {
202  d_b[i] = d_y[i];
203  }
204  }
205 }

212 void add_forward_2d_bf16(const uint16_t *a,
213  const uint16_t *b,
214  uint16_t *y,
215  int tokens,
216  int dim,
217  int aligned_dim)
218 {
219  if (!a || !b || !y || tokens <= 0 || dim <= 0) {
220  return;
221  }
222 
223  for (int t = 0; t < tokens; ++t) {
224  const uint16_t *a_row = a + (size_t)t * aligned_dim;
225  const uint16_t *b_row = b + (size_t)t * aligned_dim;
226  uint16_t *y_row = y + (size_t)t * aligned_dim;
227 
228  int d = 0;
229 
230 #if defined(__AVX512F__)
231  for (; d + 16 <= dim; d += 16) {
232  __m512 av = bf16_loadu_cvt_fp32(&a_row[d]);
233  __m512 bv = bf16_loadu_cvt_fp32(&b_row[d]);
234  __m512 yv = _mm512_add_ps(av, bv);
235  fp32_cvt_storeu_bf16(&y_row[d], yv);
236  }
237 #endif
238 
239  for (; d < dim; ++d) {
240  float af = bf16_to_float(a_row[d]);
241  float bf = bf16_to_float(b_row[d]);
242  y_row[d] = float_to_bf16(af + bf);
243  }
244  }
245 }
