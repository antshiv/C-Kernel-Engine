301 // LayerNorm backward kernel (model-agnostic), adapted from C-Transformer's
302 // backward_layernorm. Computes gradients w.r.t. input, gamma, and beta.
303 void layernorm_backward_kernel(const float *d_output, // [T×aligned_D]
304  const float *input, // [T×aligned_D]
305  const float *gamma, // [D]
306  const float *mean, // [T]
307  const float *rstd, // [T]
308  float *d_input, // [T×aligned_D]
309  float *d_gamma, // [D] (accumulated)
310  float *d_beta, // [D] (accumulated)
311  int tokens, int d_model, int aligned_embed_dim)
312 {
313  int T = tokens;
314  int D = d_model;
315  int aligned_D = aligned_embed_dim;
316 
317  // Per-token input gradients
318  for (int t = 0; t < T; ++t) {
319  float mean_t = mean[t];
320  float rstd_t = rstd[t];
321 
322  float d_y_gamma_sum = 0.0f;
323  float d_y_gamma_xhat_sum = 0.0f;
324 
325  // First pass: compute sums
326  for (int d = 0; d < D; ++d) {
327  float x = input[t * aligned_D + d];
328  float x_hat = (x - mean_t) * rstd_t;
329  float d_y = d_output[t * aligned_D + d];
330  float d_y_gamma = d_y * gamma[d];
331 
332  d_y_gamma_sum += d_y_gamma;
333  d_y_gamma_xhat_sum += d_y_gamma * x_hat;
334  }
335 
336  // Second pass: compute input gradients
337  float scale = rstd_t / (float)D;
338  for (int d = 0; d < D; ++d) {
339  float x = input[t * aligned_D + d];
340  float x_hat = (x - mean_t) * rstd_t;
341  float d_y = d_output[t * aligned_D + d];
342 
343  d_input[t * aligned_D + d] =
344  scale * ((float)D * d_y * gamma[d] - d_y_gamma_sum - x_hat * d_y_gamma_xhat_sum);
345  }
346 
347  // Zero padding for aligned dimension beyond D
348  for (int d = D; d < aligned_D; ++d) {
349  d_input[t * aligned_D + d] = 0.0f;
350  }
351  }
352 
353  // Parameter gradients (gamma, beta)
354  for (int d = 0; d < D; ++d) {
355  float gamma_grad = 0.0f;
356  float beta_grad = 0.0f;
357 
358  for (int t = 0; t < T; ++t) {
359  float x = input[t * aligned_D + d];
360  float x_hat = (x - mean[t]) * rstd[t];
361  float d_y = d_output[t * aligned_D + d];
362 
363  gamma_grad += d_y * x_hat;
364  beta_grad += d_y;
365  }
366 
367  d_gamma[d] += gamma_grad;
368  d_beta[d] += beta_grad;
369  }
370 }
