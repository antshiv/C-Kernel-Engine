33 /* GELU activation: 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3))) */
34 static inline float gelu_scalar(float x)
35 {
36  const float c = 0.7978845608f; /* sqrt(2/pi) */
37  const float k = 0.044715f;
38  float x3 = x * x * x;
39  return 0.5f * x * (1.0f + tanhf(c * (x + k * x3)));
40 }
41 
42 #if defined(__AVX512F__)
43 /* Vectorized GELU using polynomial approximation of tanh */
44 static inline __m512 gelu_avx512(__m512 x)
45 {
46  const __m512 c = _mm512_set1_ps(0.7978845608f);
47  const __m512 k = _mm512_set1_ps(0.044715f);
48  const __m512 half = _mm512_set1_ps(0.5f);
49  const __m512 one = _mm512_set1_ps(1.0f);
50 
51  __m512 x2 = _mm512_mul_ps(x, x);
52  __m512 x3 = _mm512_mul_ps(x2, x);
53 
54  /* inner = sqrt(2/pi) * (x + 0.044715 * x^3) */
55  __m512 inner = _mm512_fmadd_ps(k, x3, x);
56  inner = _mm512_mul_ps(c, inner);
57 
58  /* tanh approximation: tanh(x) ≈ x * (27 + x^2) / (27 + 9*x^2) for small x */
59  /* For larger x, clamp to ±1 */
60  __m512 inner2 = _mm512_mul_ps(inner, inner);
61  __m512 num = _mm512_add_ps(_mm512_set1_ps(27.0f), inner2);
62  __m512 den = _mm512_fmadd_ps(_mm512_set1_ps(9.0f), inner2, _mm512_set1_ps(27.0f));
63  __m512 tanh_approx = _mm512_mul_ps(inner, _mm512_div_ps(num, den));
64 
65  /* Clamp to [-1, 1] */
66  tanh_approx = _mm512_min_ps(tanh_approx, one);
67  tanh_approx = _mm512_max_ps(tanh_approx, _mm512_set1_ps(-1.0f));
68 
69  /* 0.5 * x * (1 + tanh(...)) */
70  __m512 result = _mm512_add_ps(one, tanh_approx);
71  result = _mm512_mul_ps(half, _mm512_mul_ps(x, result));
72 
73  return result;
74 }
75 #endif

77 /**
78  * Optimized MLP Forward (BF16 weights, FP32 activations)
79  *
80  * This version:
81  * 1. Uses optimized BF16 GEMM directly (no bulk conversion)
82  * 2. Keeps activations in FP32 for GELU accuracy
83  * 3. Vectorized GELU with AVX-512
84  */
85 void mlp_token_parallel_bf16(const uint16_t *input,
86  const uint16_t *W_fc1,
87  const uint16_t *b_fc1,
88  const uint16_t *W_fc2,
89  const uint16_t *b_fc2,
90  float *fc1_output,
91  float *output,
92  int T,
93  int aligned_dim,
94  int num_threads)
95 {
96  if (!input || !W_fc1 || !b_fc1 || !W_fc2 || !b_fc2 || !fc1_output || !output) {
97  return;
98  }
99 
100  const int D = aligned_dim;
101  const int fourD = 4 * D;
102 
103  /* Convert biases to FP32 (small, one-time cost) */
104  float *bias1_f = (float *)malloc(fourD * sizeof(float));
105  float *bias2_f = (float *)malloc(D * sizeof(float));
106  if (!bias1_f || !bias2_f) {
107  free(bias1_f);
108  free(bias2_f);
109  return;
110  }
111 
112  for (int i = 0; i < fourD; ++i) {
113  bias1_f[i] = bf16_to_float(b_fc1[i]);
114  }
115  for (int i = 0; i < D; ++i) {
116  bias2_f[i] = bf16_to_float(b_fc2[i]);
117  }
118 
119  /* FC1: [T, D] x [4D, D].T -> [T, 4D] with FP32 output */
120  gemm_bf16_fp32out(input, W_fc1, bias1_f, fc1_output, T, fourD, D);
121 
122  /* GELU activation (in-place on FP32) */
123 #if defined(__AVX512F__)
124  #pragma omp parallel for
125  for (int t = 0; t < T; ++t) {
126  float *row = fc1_output + (size_t)t * fourD;
127  int j = 0;
128  for (; j <= fourD - 16; j += 16) {
129  __m512 x = _mm512_loadu_ps(row + j);
130  __m512 y = gelu_avx512(x);
131  _mm512_storeu_ps(row + j, y);
132  }
133  for (; j < fourD; ++j) {
134  row[j] = gelu_scalar(row[j]);
135  }
136  }
137 #else
138  for (int t = 0; t < T; ++t) {
139  for (int j = 0; j < fourD; ++j) {
140  fc1_output[t * fourD + j] = gelu_scalar(fc1_output[t * fourD + j]);
141  }
142  }
143 #endif
144 
145  /* FC2: [T, 4D] x [D, 4D].T -> [T, D]
146  * Need to convert fc1_output to BF16 for BF16 GEMM, or use FP32 GEMM
147  * For now, use a hybrid approach: convert fc1_output to BF16 temp buffer
148  */
149  uint16_t *fc1_bf16 = (uint16_t *)malloc((size_t)T * fourD * sizeof(uint16_t));
150  if (!fc1_bf16) {
151  free(bias1_f);
152  free(bias2_f);
153  return;
154  }
155 
156  /* Convert FP32 activations back to BF16 */
157 #if defined(__AVX512F__)
158  #pragma omp parallel for
159  for (int t = 0; t < T; ++t) {
160  float *src = fc1_output + (size_t)t * fourD;
161  uint16_t *dst = fc1_bf16 + (size_t)t * fourD;
162  int j = 0;
163  for (; j <= fourD - 16; j += 16) {
164  __m512 fp32 = _mm512_loadu_ps(src + j);
165  /* Round to nearest even */
166  __m512i as_int = _mm512_castps_si512(fp32);
167  __m512i lsb = _mm512_srli_epi32(as_int, 16);
168  lsb = _mm512_and_si512(lsb, _mm512_set1_epi32(1));
169  __m512i rounding = _mm512_add_epi32(_mm512_set1_epi32(0x7FFF), lsb);
170  __m512i rounded = _mm512_add_epi32(as_int, rounding);
171  __m512i shifted = _mm512_srli_epi32(rounded, 16);
172  __m256i bf16 = _mm512_cvtepi32_epi16(shifted);
173  _mm256_storeu_si256((__m256i *)(dst + j), bf16);
174  }
175  for (; j < fourD; ++j) {
176  dst[j] = float_to_bf16(src[j]);
177  }
178  }
179 #else
180  for (size_t i = 0; i < (size_t)T * fourD; ++i) {
181  fc1_bf16[i] = float_to_bf16(fc1_output[i]);
182  }
183 #endif
184 
185  /* FC2: BF16 GEMM with FP32 output */
186  gemm_bf16_fp32out(fc1_bf16, W_fc2, bias2_f, output, T, D, fourD);
187 
188  free(fc1_bf16);
189  free(bias1_f);
190  free(bias2_f);
191 }

268  free(bias2_f);
