1 /**
2  * RoPE (Rotary Position Embedding) Kernels
3  *
4  * Applies rotary position embeddings to query and key vectors.
5  * Used by Llama, SmolLM, and most modern transformer architectures.
6  *
7  * Math:
8  * For each position m and dimension pair (2i, 2i+1):
9  * x'[2i] = x[2i] * cos(m * θ_i) - x[2i+1] * sin(m * θ_i)
10  * x'[2i+1] = x[2i] * sin(m * θ_i) + x[2i+1] * cos(m * θ_i)
11  *
12  * Where θ_i = 1 / (base^(2i/d)), typically base=10000.
13  *
14  * Layout:
15  * x: [num_heads, num_tokens, head_dim] head-major
16  * cos_cache, sin_cache: [max_seq_len, head_dim/2] precomputed
17  */

26 // Precompute cos/sin cache for given sequence length and head dimension.
27 // cos_cache, sin_cache: [max_seq_len, head_dim/2]
28 void rope_precompute_cache(float *cos_cache,
29  float *sin_cache,
30  int max_seq_len,
31  int head_dim,
32  float base)
33 {
34  int half_dim = head_dim / 2;
35 
36  for (int pos = 0; pos < max_seq_len; ++pos) {
37  for (int i = 0; i < half_dim; ++i) {
38  float theta = 1.0f / powf(base, (float)(2 * i) / (float)head_dim);
39  float angle = (float)pos * theta;
40  cos_cache[pos * half_dim + i] = cosf(angle);
41  sin_cache[pos * half_dim + i] = sinf(angle);
42  }
43  }
44 }

46 // Apply RoPE to a single head's Q or K tensor in-place.
47 // x: [num_tokens, head_dim] for one head
48 // cos_cache, sin_cache: [max_seq_len, head_dim/2]
49 // pos_offset: starting position (for KV cache continuation)
50 static inline void rope_apply_head(float *x,
51  const float *cos_cache,
52  const float *sin_cache,
53  int num_tokens,
54  int head_dim,
55  int pos_offset)
56 {
57  int half_dim = head_dim / 2;
58 
59  for (int t = 0; t < num_tokens; ++t) {
60  int pos = pos_offset + t;
61  const float *cos_row = cos_cache + pos * half_dim;
62  const float *sin_row = sin_cache + pos * half_dim;
63  float *x_row = x + t * head_dim;
64 
65  for (int i = 0; i < half_dim; ++i) {
66  float x0 = x_row[2 * i];
67  float x1 = x_row[2 * i + 1];
68  float c = cos_row[i];
69  float s = sin_row[i];
70 
71  x_row[2 * i] = x0 * c - x1 * s;
72  x_row[2 * i + 1] = x0 * s + x1 * c;
73  }
74  }
75 }

98 // RoPE backward: inverse rotation (rotate by -θ).
99 // Since cos(-θ) = cos(θ) and sin(-θ) = -sin(θ), the inverse is:
100 // d_x[2i] = d_x'[2i] * cos(θ) + d_x'[2i+1] * sin(θ)
101 // d_x[2i+1] = -d_x'[2i] * sin(θ) + d_x'[2i+1] * cos(θ)
102 //
103 // d_x: [num_heads, num_tokens, head_dim] - gradient w.r.t. input (output)
104 // d_out: [num_heads, num_tokens, head_dim] - gradient from upstream (input)
105 // cos_cache, sin_cache: [max_seq_len, head_dim/2]
106 void rope_backward(const float *d_out,
107  float *d_x,
108  const float *cos_cache,
109  const float *sin_cache,
110  int num_heads,
111  int num_tokens,
112  int head_dim,
113  int pos_offset)
114 {
115  size_t head_stride = (size_t)num_tokens * (size_t)head_dim;
116  int half_dim = head_dim / 2;
117 
118  for (int h = 0; h < num_heads; ++h) {
119  for (int t = 0; t < num_tokens; ++t) {
120  int pos = pos_offset + t;
121  const float *cos_row = cos_cache + pos * half_dim;
122  const float *sin_row = sin_cache + pos * half_dim;
123 
124  size_t idx = h * head_stride + t * head_dim;
125  const float *d_out_row = d_out + idx;
126  float *d_x_row = d_x + idx;
127 
128  for (int i = 0; i < half_dim; ++i) {
129  float d0 = d_out_row[2 * i];
130  float d1 = d_out_row[2 * i + 1];
131  float c = cos_row[i];
132  float s = sin_row[i];
133 
134  // Inverse rotation: rotate by -θ
135  d_x_row[2 * i] = d0 * c + d1 * s;
136  d_x_row[2 * i + 1] = -d0 * s + d1 * c;
137  }
138  }
139  }
140 }
