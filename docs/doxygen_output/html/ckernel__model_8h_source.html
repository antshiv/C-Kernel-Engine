61 /**

84 /**

86  *
87  * Expected layout in the file (float32, little-endian), in the same order
88  * as layout_transformer_from_ir assigns weight offsets:
89  *
90  * 1) Token embeddings [vocab_size × hidden_size]
91  * 2) Pos embeddings [context_window × hidden_size]
92  * 3) For each layer L = 0..num_layers-1:
93  * - LN1 gamma [hidden_size]
94  * - LN1 beta [hidden_size]
95  * - QKV weight [hidden_size × 3*hidden_size]
96  * - QKV bias [3*hidden_size]
97  * - Attn proj W [hidden_size × hidden_size]
98  * - Attn proj b [hidden_size]
99  * - FC1 weight [hidden_size × intermediate_size]
100  * - FC1 bias [intermediate_size]
101  * - FC2 weight [intermediate_size × hidden_size]
102  * - FC2 bias [hidden_size]
103  * 4) Final LN gamma [hidden_size]
104  * 5) Final LN beta [hidden_size]
105  * 6) LM head weight [vocab_size × hidden_size]
106  *
107  * Activation buffers (embedded_input_offset, final_output_offset, logits_offset)
108  * are NOT populated by this loader.
109  *
110  * Returns 0 on success, non-zero on failure.
