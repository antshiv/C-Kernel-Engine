42 static void gemm_bf16_scalar(const uint16_t *A,
43  const uint16_t *B,
44  const uint16_t *bias,
45  uint16_t *C,
46  int M, int N, int K)
47 {

68 /* Convert 16 BF16 values to 16 FP32 values */
69 static inline __m512 bf16x16_to_fp32(__m256i bf16_vec)
70 {
71  /* BF16 to FP32: shift left by 16 bits */
72  __m512i as_int = _mm512_cvtepu16_epi32(bf16_vec);
73  __m512i shifted = _mm512_slli_epi32(as_int, 16);
74  return _mm512_castsi512_ps(shifted);
75 }

77 /* Convert 16 FP32 values to 16 BF16 values (with rounding) */
78 static inline __m256i fp32x16_to_bf16(__m512 fp32_vec)
79 {
80  /* Round to nearest even, then truncate */
81  __m512i as_int = _mm512_castps_si512(fp32_vec);
82  __m512i lsb = _mm512_srli_epi32(as_int, 16);
83  lsb = _mm512_and_si512(lsb, _mm512_set1_epi32(1));
84  __m512i rounding = _mm512_add_epi32(_mm512_set1_epi32(0x7FFF), lsb);
85  __m512i rounded = _mm512_add_epi32(as_int, rounding);
86  __m512i shifted = _mm512_srli_epi32(rounded, 16);
87  return _mm512_cvtepi32_epi16(shifted);
88 }

90 /* BF16 dot product: 16 pairs, accumulate to FP32 */
91 static inline __m512 bf16_dot16(__m256i a_bf16, __m256i b_bf16, __m512 acc)
92 {
93  __m512 a_fp32 = bf16x16_to_fp32(a_bf16);
94  __m512 b_fp32 = bf16x16_to_fp32(b_bf16);
95  return _mm512_fmadd_ps(a_fp32, b_fp32, acc);
96 }

98 /* ==========================================================================
99  * AVX-512 Vectorized GEMM (using AVX-512F, works everywhere)
100  * C[M,N] = A[M,K] @ B[N,K].T
101  * ========================================================================== */
102 static void gemm_bf16_avx512(const uint16_t *A,
103  const uint16_t *B,
104  const uint16_t *bias,
105  uint16_t *C,
106  int M, int N, int K)
107 {
108  #pragma omp parallel for schedule(dynamic)
109  for (int i = 0; i < M; ++i) {
110  const uint16_t *a_row = A + (size_t)i * K;
111 
112  for (int j = 0; j < N; ++j) {
113  const uint16_t *b_row = B + (size_t)j * K;
114 
115  /* Initialize accumulator */
116  __m512 sum_vec = _mm512_setzero_ps();
117 
118  /* Vectorized inner loop: process 16 elements at a time */
119  int k = 0;
120  for (; k <= K - 16; k += 16) {
121  __m256i a_bf16 = _mm256_loadu_si256((const __m256i *)(a_row + k));
122  __m256i b_bf16 = _mm256_loadu_si256((const __m256i *)(b_row + k));
123  sum_vec = bf16_dot16(a_bf16, b_bf16, sum_vec);
124  }
125 
126  /* Horizontal sum */
127  float sum = _mm512_reduce_add_ps(sum_vec);
128 
129  /* Scalar tail */
130  for (; k < K; ++k) {
131  sum += bf16_to_float(a_row[k]) * bf16_to_float(b_row[k]);
132  }
133 
134  /* Add bias */
135  if (bias) {
136  sum += bf16_to_float(bias[j]);
137  }
138 
139  C[(size_t)i * N + j] = float_to_bf16(sum);
140  }
141  }
142 }

144 /* ==========================================================================
145  * Cache-Blocked AVX-512 GEMM
146  * Better memory access pattern for large matrices
147  * ========================================================================== */
148 static void gemm_bf16_blocked_avx512(const uint16_t *A,
149  const uint16_t *B,
150  const uint16_t *bias,
151  uint16_t *C,
152  int M, int N, int K)
153 {
154  /* Initialize C with bias */
155  #pragma omp parallel for
156  for (int i = 0; i < M; ++i) {
157  for (int j = 0; j < N; ++j) {
158  float b = bias ? bf16_to_float(bias[j]) : 0.0f;
159  C[(size_t)i * N + j] = float_to_bf16(b);
160  }
161  }
162 
163  /* Blocked GEMM */
164  #pragma omp parallel for collapse(2) schedule(dynamic)
165  for (int ii = 0; ii < M; ii += BLK_M) {
166  for (int jj = 0; jj < N; jj += BLK_N) {
167  int i_end = ck_min_i(ii + BLK_M, M);
168  int j_end = ck_min_i(jj + BLK_N, N);
169 
170  /* Local FP32 accumulator for this block */
171  float acc[BLK_M][BLK_N];
172  for (int i = 0; i < BLK_M; ++i) {
173  for (int j = 0; j < BLK_N; ++j) {
174  acc[i][j] = 0.0f;
175  }
176  }
177 
178  /* K-dimension blocking */
179  for (int kk = 0; kk < K; kk += BLK_K) {
180  int k_end = ck_min_i(kk + BLK_K, K);
181 
182  for (int i = ii; i < i_end; ++i) {
183  const uint16_t *a_row = A + (size_t)i * K;
184  int local_i = i - ii;
185 
186  for (int j = jj; j < j_end; ++j) {
187  const uint16_t *b_row = B + (size_t)j * K;
188  int local_j = j - jj;
189 
190  __m512 sum_vec = _mm512_setzero_ps();
191 
192  int k = kk;
193  for (; k <= k_end - 16; k += 16) {
194  __m256i a_bf16 = _mm256_loadu_si256((const __m256i *)(a_row + k));
195  __m256i b_bf16 = _mm256_loadu_si256((const __m256i *)(b_row + k));
196  sum_vec = bf16_dot16(a_bf16, b_bf16, sum_vec);
197  }
198 
199  float partial = _mm512_reduce_add_ps(sum_vec);
200  for (; k < k_end; ++k) {
201  partial += bf16_to_float(a_row[k]) * bf16_to_float(b_row[k]);
202  }
203 
204  acc[local_i][local_j] += partial;
205  }
206  }
207  }
208 
209  /* Write accumulated results back */
210  for (int i = ii; i < i_end; ++i) {
211  for (int j = jj; j < j_end; ++j) {
212  float old_val = bf16_to_float(C[(size_t)i * N + j]);
213  float new_val = old_val + acc[i - ii][j - jj];
214  C[(size_t)i * N + j] = float_to_bf16(new_val);
215  }
216  }
217  }
218  }
219 }

234 static void gemm_bf16_native(const uint16_t *A,
235  const uint16_t *B,
236  const uint16_t *bias,
237  uint16_t *C,
238  int M, int N, int K)
239 {
240  #pragma omp parallel for schedule(dynamic)
241  for (int i = 0; i < M; ++i) {
242  for (int j = 0; j < N; ++j) {
243  /* Initialize accumulator */
244  __m512 sum_vec = _mm512_setzero_ps();
245 
246  /* Native BF16 dot product: 32 pairs per instruction! */
247  int k = 0;
248  for (; k <= K - 32; k += 32) {
249  __m512bh a_vec = load_bf16x32(A + (size_t)i * K + k);
250  __m512bh b_vec = load_bf16x32(B + (size_t)j * K + k);
251  sum_vec = _mm512_dpbf16_ps(sum_vec, a_vec, b_vec);
252  }
253 
254  float sum = _mm512_reduce_add_ps(sum_vec);
255 
256  /* Scalar tail */
257  for (; k < K; ++k) {
258  sum += bf16_to_float(A[(size_t)i * K + k]) *
259  bf16_to_float(B[(size_t)j * K + k]);
260  }
261 
262  if (bias) {
263  sum += bf16_to_float(bias[j]);
264  }
265 
266  C[(size_t)i * N + j] = float_to_bf16(sum);
267  }
268  }

368 /* gemm_nn_bf16: C = A @ B (no transpose), for dL/dX computation */
369 void gemm_nn_bf16(const uint16_t *A,
370  const uint16_t *B,
371  const uint16_t *bias,
372  uint16_t *C,
373  int M, int N, int K)
374 {
375  if (!A || !B || !C || M <= 0 || N <= 0 || K <= 0) {
376  return;
377  }
378 
379 #if defined(__AVX512F__)
380  #pragma omp parallel for
381  for (int i = 0; i < M; ++i) {
382  /* Initialize row with bias */
383  int j = 0;
384  for (; j <= N - 16; j += 16) {
385  __m512 b_vec = bias ? bf16x16_to_fp32(_mm256_loadu_si256((const __m256i *)(bias + j)))
386  : _mm512_setzero_ps();
387  __m256i out = fp32x16_to_bf16(b_vec);
388  _mm256_storeu_si256((__m256i *)(C + (size_t)i * N + j), out);
389  }
390  for (; j < N; ++j) {
391  float b = bias ? bf16_to_float(bias[j]) : 0.0f;
392  C[(size_t)i * N + j] = float_to_bf16(b);
393  }
394 
395  /* Accumulate: C[i,:] += A[i,k] * B[k,:] */
396  for (int k = 0; k < K; ++k) {
397  float a_val = bf16_to_float(A[(size_t)i * K + k]);
398  __m512 a_broadcast = _mm512_set1_ps(a_val);
399 
400  j = 0;
401  for (; j <= N - 16; j += 16) {
402  __m256i b_bf16 = _mm256_loadu_si256((const __m256i *)(B + (size_t)k * N + j));
403  __m512 b_fp32 = bf16x16_to_fp32(b_bf16);
404 
405  __m256i c_bf16 = _mm256_loadu_si256((const __m256i *)(C + (size_t)i * N + j));
406  __m512 c_fp32 = bf16x16_to_fp32(c_bf16);
407 
408  c_fp32 = _mm512_fmadd_ps(a_broadcast, b_fp32, c_fp32);
409 
410  __m256i c_out = fp32x16_to_bf16(c_fp32);
411  _mm256_storeu_si256((__m256i *)(C + (size_t)i * N + j), c_out);
412  }
413  for (; j < N; ++j) {
414  float c_val = bf16_to_float(C[(size_t)i * N + j]);
415  c_val += a_val * bf16_to_float(B[(size_t)k * N + j]);
416  C[(size_t)i * N + j] = float_to_bf16(c_val);
417  }
418  }
419  }
420 #else
421  /* Scalar fallback */
422  for (int i = 0; i < M; ++i) {
423  for (int j = 0; j < N; ++j) {
424  float sum = bias ? bf16_to_float(bias[j]) : 0.0f;
425  for (int k = 0; k < K; ++k) {
426  sum += bf16_to_float(A[(size_t)i * K + k]) *
427  bf16_to_float(B[(size_t)k * N + j]);
428  }
429  C[(size_t)i * N + j] = float_to_bf16(sum);
430  }
431  }
432 #endif

435 /* gemm_tn_bf16: C = A.T @ B, for dL/dW computation */
436 void gemm_tn_bf16(const uint16_t *A,
437  const uint16_t *B,
438  const uint16_t *bias,
439  uint16_t *C,
440  int M, int N, int K)
441 {
442  if (!A || !B || !C || M <= 0 || N <= 0 || K <= 0) {
443  return;
444  }
445 
446  /* A is [K x M], we want A.T which is [M x K] */
447  /* B is [K x N] */
448  /* C is [M x N] */
449 
450 #if defined(__AVX512F__)
451  /* Initialize C with bias */
452  #pragma omp parallel for
453  for (int i = 0; i < M; ++i) {
454  for (int j = 0; j < N; ++j) {
455  float b = bias ? bf16_to_float(bias[j]) : 0.0f;
456  C[(size_t)i * N + j] = float_to_bf16(b);
457  }
458  }
459 
460  /* Accumulate: C[i,j] += sum_k A[k,i] * B[k,j] */
461  #pragma omp parallel for
462  for (int i = 0; i < M; ++i) {
463  for (int j = 0; j < N; ++j) {
464  __m512 sum_vec = _mm512_setzero_ps();
465 
466  int k = 0;
467  for (; k <= K - 16; k += 16) {
468  /* Gather A[k:k+16, i] - strided access */
469  __m512 a_fp32 = _mm512_setzero_ps();
470  for (int kk = 0; kk < 16; ++kk) {
471  float val = bf16_to_float(A[(size_t)(k + kk) * M + i]);
472  a_fp32 = _mm512_mask_mov_ps(a_fp32, 1 << kk, _mm512_set1_ps(val));
473  }
474 
475  /* Note: B has stride N, so we need to gather element by element */
476  __m512 b_fp32 = _mm512_setzero_ps();
477  for (int kk = 0; kk < 16; ++kk) {
478  float val = bf16_to_float(B[(size_t)(k + kk) * N + j]);
479  b_fp32 = _mm512_mask_mov_ps(b_fp32, 1 << kk, _mm512_set1_ps(val));
480  }
481 
482  sum_vec = _mm512_fmadd_ps(a_fp32, b_fp32, sum_vec);
483  }
484 
485  float sum = _mm512_reduce_add_ps(sum_vec);
486 
487  for (; k < K; ++k) {
488  sum += bf16_to_float(A[(size_t)k * M + i]) *
489  bf16_to_float(B[(size_t)k * N + j]);
490  }
491 
492  float old_val = bf16_to_float(C[(size_t)i * N + j]);
493  C[(size_t)i * N + j] = float_to_bf16(old_val + sum);
494  }
495  }
496 #else
497  for (int i = 0; i < M; ++i) {
498  for (int j = 0; j < N; ++j) {
499  float sum = bias ? bf16_to_float(bias[j]) : 0.0f;
500  for (int k = 0; k < K; ++k) {
501  sum += bf16_to_float(A[(size_t)k * M + i]) *
502  bf16_to_float(B[(size_t)k * N + j]);
503  }
504  C[(size_t)i * N + j] = float_to_bf16(sum);
505  }
506  }
507 #endif
508 }
