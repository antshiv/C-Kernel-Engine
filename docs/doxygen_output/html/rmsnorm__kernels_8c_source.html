10 static inline float hsum256_ps_rmsnorm(__m256 v) {
11  // Sum upper and lower 128-bit lanes
12  __m128 hi = _mm256_extractf128_ps(v, 1);
13  __m128 lo = _mm256_castps256_ps128(v);
14  __m128 sum128 = _mm_add_ps(lo, hi);
15  // Horizontal add within 128-bit lane
16  sum128 = _mm_hadd_ps(sum128, sum128);
17  sum128 = _mm_hadd_ps(sum128, sum128);
18  return _mm_cvtss_f32(sum128);

22 // RMSNorm forward:
23 // For each token t:
24 // r = sqrt( (1/D) * sum_i x_i^2 + eps )
25 // rstd = 1 / r
26 // x_hat_i = x_i * rstd
27 // y_i = gamma_i * x_hat_i
28 //
29 // We cache rstd per token for use in backward.
30 void rmsnorm_forward(const float *input,
31  const float *gamma,
32  float *output,
33  float *rstd_cache,
34  int tokens,
35  int d_model,
36  int aligned_embed_dim,
37  float eps)

41  int aligned = aligned_embed_dim;

45  float *y = output + (size_t)t * aligned;

47 #if defined(__AVX512F__)
48  // AVX-512: Process 16 floats at a time
49  __m512 sum_sq_vec = _mm512_setzero_ps();
50  int d = 0;
51 
52  // Vectorized sum of squares
53  for (; d + 16 <= D; d += 16) {
54  __m512 xv = _mm512_loadu_ps(&x[d]);
55  sum_sq_vec = _mm512_fmadd_ps(xv, xv, sum_sq_vec);
56  }
57  float sum_sq = _mm512_reduce_add_ps(sum_sq_vec);
58 
59  // Handle remaining elements
60  for (; d < D; ++d) {
61  sum_sq += x[d] * x[d];
62  }
63 
64  float mean_sq = sum_sq / (float)D;
65  float rstd = 1.0f / sqrtf(mean_sq + eps);
66  rstd_cache[t] = rstd;
67 
68  // Apply normalization and scale (vectorized)
69  __m512 rstd_vec = _mm512_set1_ps(rstd);
70  d = 0;
71  for (; d + 16 <= D; d += 16) {
72  __m512 xv = _mm512_loadu_ps(&x[d]);
73  __m512 gv = _mm512_loadu_ps(&gamma[d]);
74  __m512 x_hat = _mm512_mul_ps(xv, rstd_vec);
75  __m512 yv = _mm512_mul_ps(x_hat, gv);
76  _mm512_storeu_ps(&y[d], yv);
77  }
78  // Handle remaining elements
79  for (; d < D; ++d) {
80  y[d] = x[d] * rstd * gamma[d];
81  }
82 
83 #elif defined(__AVX__)
84  // AVX: Process 8 floats at a time
85  __m256 sum_sq_vec = _mm256_setzero_ps();
86  int d = 0;
87

91  __m256 xv_sq = _mm256_mul_ps(xv, xv);
92  sum_sq_vec = _mm256_add_ps(sum_sq_vec, xv_sq);
93  }
94  float sum_sq = hsum256_ps_rmsnorm(sum_sq_vec);
95 
96  // Handle remaining elements
97  for (; d < D; ++d) {
98  sum_sq += x[d] * x[d];
99  }

102  float rstd = 1.0f / sqrtf(mean_sq + eps);
103  rstd_cache[t] = rstd;
104

111  __m256 x_hat = _mm256_mul_ps(xv, rstd_vec);
112  __m256 yv = _mm256_mul_ps(x_hat, gv);
113  _mm256_storeu_ps(&y[d], yv);
114  }
115  // Handle remaining elements
116  for (; d < D; ++d) {
117  y[d] = x[d] * rstd * gamma[d];
118  }

121  // Scalar fallback
122  double sum_sq = 0.0;
123  for (int d = 0; d < D; ++d) {
124  double v = (double)x[d];

126  }

146 // RMSNorm backward:
147 // Given dY, X, gamma, and cached rstd per token, compute:
148 // dX and dGamma.
149 //
150 // From derivation:
151 // x_hat_i = x_i * rstd_t
152 // m = (1/D) * sum_j (dY_j * gamma_j * x_hat_j)
153 // dX_i = rstd_t * (dY_i * gamma_i - x_hat_i * m)
154 // dGamma_i = sum_t (dY_i * x_hat_i)
155 //
156 // We do not include a beta parameter for RMSNorm here.
157 void rmsnorm_backward(const float *d_output,
158  const float *input,
159  const float *gamma,

164  int d_model,
165  int aligned_embed_dim)

168  int D = d_model;
169  int aligned = aligned_embed_dim;
170 
171  // Zero parameter gradients
172 #if defined(__AVX512F__)
173  {
174  int d = 0;
175  for (; d + 16 <= D; d += 16) {
176  _mm512_storeu_ps(&d_gamma[d], _mm512_setzero_ps());
177  }
178  for (; d < D; ++d) {
179  d_gamma[d] = 0.0f;
180  }
181  }
182 #elif defined(__AVX__)
183  {

187  }

190  }
191  }

208  __m512 sum_vec = _mm512_setzero_ps();
209  int d = 0;
210

221 
222  // Handle remaining elements

226  }

231  d = 0;
232  for (; d + 16 <= D; d += 16) {

234  __m512 dyv = _mm512_loadu_ps(&dY[d]);
235  __m512 gv = _mm512_loadu_ps(&gamma[d]);
236  __m512 dgv = _mm512_loadu_ps(&d_gamma[d]);

242  __m512 xhat_m = _mm512_mul_ps(x_hat, m_vec);
243  __m512 diff = _mm512_sub_ps(dy_g, xhat_m);
